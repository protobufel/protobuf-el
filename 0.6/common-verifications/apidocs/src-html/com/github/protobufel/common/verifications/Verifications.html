028package com.github.protobufel.common.verifications;
029
030import org.eclipse.jdt.annotation.Nullable;
031
032public final class Verifications {
033  private Verifications() {
034  }
035
036  public static <T> T assertNonNull(final @Nullable T value) {
037    if (value == null) {
038      throw new AssertionError();
039    }
040    
041    return value;
042  }
043
044  public static <T> T assertNonNull(final @Nullable T value, final @Nullable String msg) {
045    if (value == null) {
046      throw new AssertionError(msg);
047    }
048    
049    return value;
050  }
051
052  public static <E, T extends Iterable<E>> T verifyNonNullIterable(final @Nullable T values) {
053    if (values == null) {
054      throw new NullPointerException();
055    }
056
057    for (E value : values) {
058      if (value == null) {
059        throw new NullPointerException();
060      }
061    }
062
063    return values;
064  }
065  
066  public static <E, T extends Iterable<E>> T verifyNonNullIterable(final @Nullable T values,
067      final @Nullable String msg) {
068    if (values == null) {
069      throw new NullPointerException();
070    }
071
072    for (E value : values) {
073      if (value == null) {
074        throw new NullPointerException(msg);
075      }
076    }
077
078    return values;
079  }
080
081  public static <E, T extends Iterable<E>> T verifyNonNullElements(final T values) {
082    for (E value : values) {
083      if (value == null) {
084        throw new NullPointerException();
085      }
086    }
087
088    return values;
089  }
090
091  public static <E, T extends Iterable<E>> T verifyNonNullElements(final T values,
092      final @Nullable String msg) {
093    for (E value : values) {
094      if (value == null) {
095        throw new NullPointerException(msg);
096      }
097    }
098
099    return values;
100  }
101
102  public static <T> T verifyNonNull(final @Nullable T value) {
103    if (value == null) {
104      throw new NullPointerException();
105    }
106
107    return value;
108  }
109
110  public static <T> T verifyNonNull(final @Nullable T value, final @Nullable String msg) {
111    if (value == null) {
112      throw new NullPointerException(msg);
113    }
114
115    return value;
116  }
117
118  public static <T> T verifyArgument(final boolean condition, final @Nullable T value) {
119    if (!condition) {
120      throw new IllegalArgumentException();
121    } else if (value == null) {
122      throw new NullPointerException();
123    }
124
125    return value;
126  }
127
128  public static <T> T verifyArgument(final boolean condition, final @Nullable T value,
129      final @Nullable String msg) {
130    if (!condition) {
131      throw new IllegalArgumentException(msg);
132    } else if (value == null) {
133      throw new NullPointerException();
134    }
135
136    return value;
137  }
138
139  public static boolean verifyCondition(final boolean condition) {
140    if (!condition) {
141      throw new IllegalArgumentException();
142    }
143
144    return condition;
145  }
146
147  public static boolean verifyCondition(final boolean condition, final @Nullable String msg) {
148    if (!condition) {
149      throw new IllegalArgumentException(msg);
150    }
151
152    return condition;
153  }
154  
155  public static <T> T nonNullValue(final @Nullable T value, final T defaultValue) {
156    return (value == null) ? defaultValue : value;
157  }
158}
