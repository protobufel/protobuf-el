044/**
045 * Support for EL-friendly ProtoBuf Attribute/Builder/Message Lists. Provides a special object when
046 * a repeated field is requested on a protoBuf Builder. This object has the following methods and
047 * EL/JavaBean properties:
048 * <ol>
049 * <li>getMessages()/messages - immutable list/view of Messages
050 * <li>getBuilders()/builders - immutable list/view of Builders
051 * <li>add() - adds a new default/empty Builder and returns it
052 * <li>addMessage(index, Message) - adds the Message and returns its Builder
053 * <li>addBuilder(index, Builder)
054 * <li>setMessage(index, Message)
055 * <li>setBuilder(index, Builder)
056 * <li>remove(index)
057 * <li>clear()
058 * <li>getParent()/parent
059 * <li>getChanged()/changed
060 * <li>getChangedIndex()/changedIndex
061 * </ol>
062 * <p>
063 * Sub-alternative:
064 * <ol>
065 * <li>getMessages()/messages - immutable list/view of Messages
066 * <li>getBuilders()/builders - immutable list/view of Builders
067 * <li>add() - adds a new default/empty Builder and returns it
068 * <li>add(index, Message) - adds the Message and returns back to the object
069 * <li>add(index, Builder)
070 * <li>set(index, Message)
071 * <li>set(index, Builder)
072 * <li>remove(index)
073 * <li>clear()
074 * <li>getParent()/parent
075 * <li>getChanged()/changed
076 * <li>getChangedIndex()/changedIndex
077 * </ol>
078 * Requires handling overloaded methods, but is more concise and consistent with attribute lists
079 * which have the following:
080 * <ol>
081 * <li>add() - adds the default value for the type
082 * <li>add(index, value) - adds the value and returns back to the object
083 * <li>set(index, value)
084 * <li>remove(index)
085 * <li>clear()
086 * <li>getParent()/parent
087 * <li>getChanged()/changed
088 * <li>getChangedIndex()/changedIndex
089 * </ol>
090 * <p>
091 * As an alternative, the above features could have been implemented on a top of the returned
092 * attribute, Builder, or Message list. However, the current EL collection operations don't support
093 * mutation, so this would not be very beneficial, apart from being more concise. Also, this would
094 * confuse what type of collection being returned - for a Message base object it must be a list of
095 * Messages, but for Builder base - could be either a list of Builders or Messages, and how to get
096 * the other type? And the last, it would complicate implementation a little bit.
097 * <p>
098 * The provided main approach leaves door open for incorporation of the mutation methods directly
099 * into the lists.
100 *
101 * @author protobufel@gmail.com David Tesler
102 *
103 */
104public class ProtoLists {
105  private ProtoLists() {}

213  public static final Class<?> getFieldClass(final FieldDescriptor field,
214      final boolean isRepeatedAsList) {
215    if (isRepeatedAsList && field.isRepeated()) {
216      // return IRepeatedFieldValueBuilder.class;
217      return List.class;
218    }
219
220    switch (field.getJavaType()) {
221      case BOOLEAN:
222        return Boolean.class;
223      case DOUBLE:
224        return Double.class;
225      case FLOAT:
226        return Float.class;
227      case INT:
228        return Integer.class;
229      case LONG:
230        return Long.class;
231      case STRING:
232        return String.class;
233      case BYTE_STRING:
234        return String.class;
235      case ENUM:
236        // return String.getClass();
237        // return field.getEnumType().getClass();
238        return Object.class;
239      case MESSAGE:
240        // return field.getMessageType().getClass();
241        return MessageOrBuilder.class;
242      default:
243        return null;
244    }
245  }

247  public static final Object verifyAndConvertField(final Object val, final FieldDescriptor field) {
248    switch (field.getJavaType()) {
249      case MESSAGE:
250        if (field.isRepeated()) {
251          final List<?> list = List.class.cast(val);
252          final List<Message> messages = new ArrayList<Message>(list.size());
253
254          for (final Object obj : list) {
255            messages.add(verifyAndConvertToMessageIfBuilder(field, obj));
256          }
257
258          return messages;
259        } else {
260          return verifyAndConvertToMessageIfBuilder(field, val);
261        }
262      case ENUM:
263        if (field.isRepeated()) {
264          final List<?> list = List.class.cast(val);
265          final List<EnumValueDescriptor> enums = new ArrayList<EnumValueDescriptor>(list.size());
266
267          for (final Object obj : list) {
268            enums.add(MessageAdapter.getEnumValue(field, obj));
269          }
270
271          return enums;
272        } else {
273          return MessageAdapter.getEnumValue(field, val);
274        }
275      default:
276        return val;
277    }
278  }

280  public static final Object verifyAndConvertFieldSingleValue(final Object val,
281      final FieldDescriptor field) {
282    if (val == null) {
283      throw new NullPointerException();
284    } else if (val instanceof Iterable) {
285      throw new IllegalArgumentException("must be a single value, not Iterable");
286    }
287
288    switch (field.getJavaType()) {
289      case MESSAGE:
290        return verifyAndConvertToMessageIfBuilder(field, val);
291      case ENUM:
292        return MessageAdapter.getEnumValue(field, val);
293      default:
294        return val;
295    }
296  }

298  public static final Message verifyAndConvertToMessageIfBuilder(final FieldDescriptor field,
299      final Object val) {
300    if (!(val instanceof MessageOrBuilder)) {
301      throw new IllegalArgumentException("the value should be of MessageOrBuilder type");
302    } else if (field.getMessageType() != ((MessageOrBuilder) val).getDescriptorForType()) {
303      throw new IllegalArgumentException("the value of a wrong builder type");
304    }
305
306    return val instanceof Message.Builder ? ((Message.Builder) val).buildPartial() : (Message) val;
307  }

309  public static final IRepeatedFieldValueBuilder<?> getBuilderForRepeatedField(
310      final Message.Builder builder, final FieldDescriptor field) {
311    if (field.getJavaType() == JavaType.MESSAGE) {
312      return getBuilderForRepeatedChild(builder, field);
313    } else {
314      return getBuilderForRepeatedAttribute(builder, field);
315    }
316  }

318  public static final IRepeatedFieldValueBuilder<Object> getBuilderForRepeatedAttribute(
319      final Message.Builder builder, final FieldDescriptor field) {
320    if (!field.isRepeated()) {
321      throw new IllegalArgumentException("the field must be repeated");
322    } else if (builder.getDescriptorForType() != field.getContainingType()) {
323      throw new IllegalArgumentException("the field doen't belong to this type");
324    }
325    if (field.getJavaType() == JavaType.MESSAGE) {
326      throw new IllegalArgumentException("the field must be of a non-Message type");
327    }
328
329    return new RepeatedFieldValueBuilder<Message.Builder, Object>(builder, field);
330  }
331
332  @SuppressWarnings("unchecked")
333  public static final IRepeatedFieldMessageBuilder getBuilderForRepeatedChild(
334      final Message.Builder builder, final FieldDescriptor field) {
335    if (!field.isRepeated()) {
336      throw new IllegalArgumentException("the field must be repeated");
337    } else if (field.getJavaType() != JavaType.MESSAGE) {
338      throw new IllegalArgumentException("the field must be of Message type");
339    } else if (builder.getDescriptorForType() != field.getContainingType()) {
340      throw new IllegalArgumentException("the field doen't belong to this type");
341    }
342
343    if (builder instanceof GeneratedMessage.Builder) {
344      final GeneratedMessage.Builder<?> generatedBuilder = (GeneratedMessage.Builder<?>) builder;
345      return getGeneratedRepeatedFieldBuilder(generatedBuilder, field, generatedBuilder.getClass());
346    } else if (builder instanceof IBuilder2) {
347      return new DynamicRepeatedFieldMessageBuilder((IBuilder2) builder, field);
348    } else {
349      throw new IllegalArgumentException("builder is not of a supported Builder type");
350    }
351  }

358  public static class RepeatedFieldValueBuilder<P extends Message.Builder, E> implements
359  IRepeatedFieldValueBuilder<E> {
360    protected final P parent;
361    protected final FieldDescriptor field;
362    protected int changedIndex = -1;
363
364    public RepeatedFieldValueBuilder(final P parent, final FieldDescriptor field) {
365      this.parent = parent;
366      this.field = field;
367    }
368
369    @SuppressWarnings("unchecked")
370    @Override
371    public E get(final int index) {
372      return (E) parent.getRepeatedField(field, index);
373    }
374
375    @Override
376    public int size() {
377      return parent.getRepeatedFieldCount(field);
378    }
379
380    @Override
381    public IRepeatedFieldValueBuilder<E> set(final int index, final E element) {
382      parent.setRepeatedField(field, index, element);
383      changedIndex = index;
384      return this;
385    }
386
387    @Override
388    public E add() {
389      add(size(), newInstance());
390      return getLast();
391    }
392
393    @SuppressWarnings("unchecked")
394    @Override
395    public E newInstance() {
396      return (E) field.getDefaultValue();
397    }
398
399    @Override
400    public IRepeatedFieldValueBuilder<E> add(final E element) {
401      return add(size(), element);
402    }
403
404    @Override
405    public IRepeatedFieldValueBuilder<E> add(final int index, final E element) {
406      @SuppressWarnings("unchecked")
407      final List<E> list = (List<E>) parent.getField(field);
408
409      if (list == null && index != 0 || index < 0 || list != null && index > list.size()) {
410        throw new IndexOutOfBoundsException();
411      }
412
413      final List<E> newList = list == null ? new ArrayList<E>() : new ArrayList<E>(list);
414      newList.add(index, element);
415      parent.setField(field, newList);
416      changedIndex = index;
417      return this;
418    }
419
420    @Override
421    public IRepeatedFieldValueBuilder<E> remove(final int index) {
422      @SuppressWarnings("unchecked")
423      final List<E> list = (List<E>) parent.getField(field);
424
425      if (list == null || index < 0 || index >= list.size()) {
426        throw new IndexOutOfBoundsException();
427      }
428
429      final List<E> newList = new ArrayList<E>(list);
430      newList.remove(index);
431      parent.setField(field, newList);
432      changedIndex = index;
433      return this;
434    }
435
436    @Override
437    public IRepeatedFieldValueBuilder<E> clear() {
438      parent.clearField(field);
439      changedIndex = -1;
440      return this;
441    }
442
443    @Override
444    public Message.Builder getParent() {
445      return parent;
446    }
447
448    @Override
449    public int getChangedIndex() {
450      return changedIndex;
451    }
452
453    @Override
454    public E getChanged(final int index) {
455      return changedIndex == -1 ? null : get(changedIndex);
456    }
457
458    @Override
459    public E getLast() {
460      final int size = size();
461      return size == 0 ? null : get(size - 1);
462    }
463
464    @SuppressWarnings("unchecked")
465    @Override
466    public List<? extends E> getList() {
467      return (List<? extends E>) parent.getField(field);
468    }
469
470    @Override
471    public IRepeatedFieldValueBuilder<E> addAll(final Collection<? extends E> values) {
472      for (final E value : values) {
473        add(value);
474      }
475
476      return this;
477    }
478
479    @Override
480    public FieldDescriptor getFieldDescriptor() {
481      return field;
482    }
483
484    @Override
485    public JavaType getType() {
486      return field.getJavaType();
487    }
488  }

574        addMessage(index, (Message) element);

577      }
578
579      return this;
580    }
581
582    @Override
583    public IRepeatedFieldMessageBuilder remove(final int index) {
584      removeElement(index);
585      return this;
586    }
587
588    @SuppressWarnings("unchecked")
589    @Override
590    public List<? extends Message> getList() {
591      return (List<? extends Message>) super.getList();
592    }
593
594    @Override
595    public IRepeatedFieldMessageBuilder clear() {
596      return (IRepeatedFieldMessageBuilder) super.clear();
597    }
598
599    @Override
600    public IRepeatedFieldMessageBuilder addAll(final Collection<? extends MessageOrBuilder> values) {
601      return (IRepeatedFieldMessageBuilder) super.addAll(values);
602    }
603  }
