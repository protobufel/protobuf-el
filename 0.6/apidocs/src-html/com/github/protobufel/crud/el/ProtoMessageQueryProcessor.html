082      final Descriptor type, final QueryResultListener resultListener,
083      final ValidationListener validationListener, final IDynamicMessageProvider messageProvider) {
084    this.type = type;

089        (validationListener == null) ? DEFAULT_VALIDATION_LISTENER : validationListener;

100    result = prime * result + (resultListener == null ? 0 : resultListener.hashCode());
101    result = prime * result + (validationListener == null ? 0 : validationListener.hashCode());

178  public List<Message> process(final List<? extends Message> originalList, final Descriptor type,
179      final String expression, final String emptyExpression, final Map<String, Object> beans,
180      final IDynamicMessageProvider messageProvider) {
181    if (originalList == null) {
182      throw new NullPointerException();
183    } else if (originalList.isEmpty() ? emptyExpression == null || emptyExpression.isEmpty()
184        : expression == null || expression.isEmpty()) {
185      return Collections.emptyList();
186    }
187
188    final ELProcessor elp = newELProcessor();
189
190    if (beans != null && !beans.isEmpty()) {
191      for (final Entry<String, Object> entry : beans.entrySet()) {
192        elp.defineBean(entry.getKey(), entry.getValue());
193      }
194    }
195
196    final List<Message> results = new ArrayList<Message>();
197    final List<Message> readOnlyResults = Collections.unmodifiableList(results);
198    int index = -1;
199
200    if (originalList.isEmpty()) {
201      final Object result = elp.eval(emptyExpression);
202
203      if (!collectResult(results, type, index, result, originalList)) {
204        return Collections.emptyList();
205      }
206
207      return readOnlyResults;
208    }
209
210    final List<Message> records = Collections.unmodifiableList(originalList);
211    elp.defineBean("records", records);
212    elp.defineBean("results", readOnlyResults);
213    final ELContext context = elp.getELManager().getELContext();
214    final ValueExpression processExpr = getValueExpression(context, expression, Object.class);
215
216    for (final Message message : records) {
217      elp.defineBean("record", getBuilder(message, messageProvider));
218      elp.defineBean("index", ++index);
219      final Object result = processExpr.getValue(context);
220
221      if (!collectResult(results, type, index, result, originalList)) {
222        return Collections.emptyList();
223      }
224    }
225
226    return readOnlyResults;
227  }
228
229  private boolean collectResult(final List<Message> results, final Descriptor type,
230      final int index, final Object result, final List<? extends Message> originalList) {
231    if (result == null) {
232      // noop - record is not added to the results, i.e. skipped!
233    } else if (result instanceof Iterable) {
234      // add all non-null items, properly converted, to the results
235      for (final Object element : (Iterable<?>) result) {
236        if (!collectSingleResult(results, type, index, element, originalList)) {
237          return false;
238        }
239      }
240    } else {
241      if (!collectSingleResult(results, type, index, result, originalList)) {
242        return false;
243      }
244    }
245
246    return true;
247  }
248
249  private boolean collectSingleResult(final List<Message> results, final Descriptor type,
250      final int index, final Object result, final List<? extends Message> originalList) {
251    if (result == null) {
252      return true;
253    } else if (!(result instanceof MessageOrBuilder)) {
254      throw new IllegalArgumentException(String.format(
255          "result of original record #%s is not a MessageOrBuilder", index));
256    } else if (!validationListener.validate(type, originalList.get(index),
257        (MessageOrBuilder) result)) {
258      throw new IllegalArgumentException(String.format(
259          "result of original record #%s is of the wrong message type", index));
260    }
261
262    if (result instanceof Message.Builder) {
263      results.add(((Message.Builder) result).build());
264    } else if (result instanceof Message) {
265      results.add((Message) result);
266    }
267
268    return notifyResultAdded(index);
269  }
270
271  /**
272   * Notifies resultListener when the record added.
273   *
274   * @param index originalList record index
275   * @return true - to continue (by default), false - to abort
276   */
277  protected boolean notifyResultAdded(final int index) {
278    return (resultListener == null) ? true : resultListener.resultAdded(index);
279  }

282    return new ProtoELProcessorEx();

285  protected Message.Builder getBuilder(final Message message,
286      final IDynamicMessageProvider messageProvider) {
287    if (message == null) {
288      return null;
289    } else if (message instanceof GeneratedMessage || message instanceof IBuilder2) {
290      return message.toBuilder();
291    } else {
292      return messageProvider.newBuilder(message);
293    }
294  }
295
296  private ValueExpression getValueExpression(final ELContext context, final String expression,
297      final Class<?> expectedType) {
298    return ELManager.getExpressionFactory().createValueExpression(context, bracket(expression),
299        expectedType);
300  }
301
302  private String bracket(final String expression) {
303    return "${" + expression + '}';
304  }

306  /**
307   * A query builder and EL 3.0 Language processor with ProtoBuf.
308   * 
309   * @author protobufel@gmail.com David Tesler
310   */
311  public static final class QueryBuilder implements IQueryProcessor<Message> {
312    private Descriptor type = null;
313    private String expression = null;
314    private String emptyExpression = null;
315    private QueryResultListener resultListener = null;
316    private ValidationListener validationListener = null;
317    private Map<String, Object> beans = null;
318    private IDynamicMessageProvider messageProvider = null;
319
320    /**
321     * Constructs a default QueryBuilder
322     */
323    public QueryBuilder() {
324      messageProvider = DynamicMessage.getProvider();
325    }
326
327    /**
328     * Constructs a QueryBuilder based on another QueryBuilder.
329     * <p>
330     * All the fields of the parameter builder are copied as-is but {@code  beans}, which are not
331     * set. This is done to prevent the inadvertent original builder's {@code beans} mutation. If
332     * needed, this can be accomplished explicitly via {@link #setBeans(Map)} or
333     * {@link #addBean(String, Object)} methods.
334     * 
335     * @param builder a template QueryBuilder to copy from
336     */
337    public QueryBuilder(final QueryBuilder builder) {
338      type = builder.type;
339      expression = builder.expression;
340      emptyExpression = builder.emptyExpression;
341      resultListener = builder.resultListener;
342      validationListener = builder.validationListener;
343      messageProvider = builder.messageProvider;
344      // beans = (builder.beans == null) ? null : new HashMap<String, Object>(builder.beans);
345    }
346
347    public QueryBuilder(final ProtoMessageQueryProcessor processor) {
348      type = processor.type;
349      expression = processor.expression;
350      emptyExpression = processor.emptyExpression;
351      resultListener = processor.resultListener;
352      validationListener = processor.validationListener;
353      messageProvider = processor.messageProvider;
354    }
355
356    public IDynamicMessageProvider getMessageProvider() {
357      return messageProvider;
358    }
359
360    public QueryBuilder setMessageProvider(final IDynamicMessageProvider messageProvider) {
361      this.messageProvider = messageProvider;
362      return this;
363    }
364
365    public Map<String, Object> getBeans() {
366      return beans;
367    }
368
369    /**
370     * Sets this QueryBuilder's beans to the shallow copy of the parameter.
371     * 
372     * @param beans a map of a bean name and its value to be copied over
373     */
374    public QueryBuilder setBeans(final Map<String, Object> beans) {
375      this.beans = beans == null ? null : new HashMap<String, Object>(beans);
376      return this;
377    }
378
379    public QueryBuilder addBean(final String name, final Object bean) {
380      if (beans == null) {
381        beans = new HashMap<String, Object>(beans);
382      }
383
384      beans.put(name, bean);
385      return this;
386    }
387
388    public Descriptor getType() {
389      return type;
390    }

392    public QueryBuilder setType(final Descriptor type) {
393      this.type = type;
394      return this;
395    }
396
397    public QueryResultListener getResultListener() {
398      return resultListener;
399    }
400
401    public QueryBuilder setResultListener(final QueryResultListener resultListener) {
402      this.resultListener = resultListener;
403      return this;
404    }
405
406    public ValidationListener getValidationListener() {
407      return validationListener;
408    }

410    public QueryBuilder setValidationListener(final ValidationListener validationListener) {
411      this.validationListener = validationListener;
412      return this;
413    }

416      return expression;
417    }
418
419    public QueryBuilder setExpression(final String expression) {
420      this.expression = expression;
421      return this;
422    }

424    public String getEmptyExpression() {
425      return emptyExpression;

429      this.emptyExpression = emptyExpression;

432
433    public ProtoMessageQueryProcessor build() {
434      if (messageProvider == null) {
435        throw new NullPointerException();
436      }
437
438      final ProtoMessageQueryProcessor processor =
439          new ProtoMessageQueryProcessor(expression, emptyExpression, type, resultListener,
440              validationListener, messageProvider);
441      return processor;
442    }

446      return process(originalList, beans);
447    }

451        final Map<String, Object> beans) {

453          messageProvider);
454    }
455  }

457  public static class DefaultValidationListener implements ValidationListener {
458
459    @Override
460    public boolean validate(final Descriptor type, final Message original,
461        final MessageOrBuilder result) {
462      if (type == null) {
463        return original == null ? true : original.getDescriptorForType().equals(
464            result.getDescriptorForType());
465      } else {
466        return type.equals(result.getDescriptorForType());
467      }
468    }
