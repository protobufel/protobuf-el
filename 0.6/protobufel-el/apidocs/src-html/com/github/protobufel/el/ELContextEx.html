028package com.github.protobufel.el;
029
030import javax.el.ArrayELResolver;
031import javax.el.BeanELResolver;
032import javax.el.CompositeELResolver;
033import javax.el.ELContext;
034import javax.el.ELResolver;
035import javax.el.ExpressionFactory;
036import javax.el.FunctionMapper;
037import javax.el.ListELResolver;
038import javax.el.MapELResolver;
039import javax.el.ResourceBundleELResolver;
040import javax.el.StandardELContext;
041import javax.el.StaticFieldELResolver;
042import javax.el.VariableMapper;
043
044/**
045 * A customized ELContext to be set on {@link StandardELContext}. Replaces the standard
046 * {@link BeanELResolver} with the more advanced {@link BeanELResolverEx}.
047 * <p>
048 * <strong>Usage</strong>: replace the standard {@link StandardELContext} with
049 * {@code new StandardELContext(new ELContextEx())}, or via
050 * {@code ELManager.setELContext(new ELContextEx())}.
051 *
052 * @author protobufel@gmail.com David Tesler
053 */
054public class ELContextEx extends ELContext {
055  private final ExpressionFactory factory;
056  private ELResolver elResolver;
057  private FunctionMapper functionMapper;
058  private VariableMapper variableMapper;
059  private CompositeELResolver customResolver;
060
061  public ELContextEx(final ExpressionFactory factory) {
062    this.factory = factory;
063  }
064
065  @Override
066  public ELResolver getELResolver() {
067    if (elResolver == null) {
068      final CompositeELResolver resolver = new CompositeELResolver();
069      customResolver = new CompositeELResolver();
070      resolver.add(customResolver);
071      resolver.add(factory.getStreamELResolver());
072      resolver.add(new StaticFieldELResolver());
073      resolver.add(new MapELResolver());
074      resolver.add(new ResourceBundleELResolver());
075      resolver.add(new ListELResolver());
076      resolver.add(new ArrayELResolver());
077      resolver.add(new BeanELResolverEx());
078      elResolver = resolver;
079    }
080
081    return elResolver;
082  }
083
084  public void addELResolver(final ELResolver resolver) {
085    getELResolver();
086    customResolver.add(resolver);
087  }
088
089  @Override
090  public FunctionMapper getFunctionMapper() {
091    if (functionMapper == null) {
092      init();
093    }
094
095    return functionMapper;
096  }
097
098  @Override
099  public VariableMapper getVariableMapper() {
100    if (variableMapper == null) {
101      init();
102    }
103
104    return variableMapper;
105  }
106
107  private void init() {
108    final StandardELContext context = new StandardELContext(factory);
109    functionMapper = context.getFunctionMapper();
110    variableMapper = context.getVariableMapper();
111  }
112}
