046/**
047 * A BeanELResolver with advanced handling of the bean's overloaded methods. Consults with the
048 * {@code base}'s BeanInfo, if present, to better handle overloading methods' ambiguity - the
049 * BeanInfo declared methods are in preference to the rest.
050 *
051 * @author protobufel@gmail.com David Tesler
052 */
053public class BeanELResolverEx extends BeanELResolver {
054  protected static final Comparator<Method> DEFAULT_SUITABLE_METHOD_COMPARATOR =
055      new SuitableMethodComparator();
056  private final Comparator<Method> suitableMethodComparator;
057
058  public BeanELResolverEx() {
059    suitableMethodComparator = DEFAULT_SUITABLE_METHOD_COMPARATOR;
060  }
061
062  public BeanELResolverEx(final Comparator<Method> suitableMethodComparator) {
063    this.suitableMethodComparator = suitableMethodComparator;
064  }
065
066  /**
067   * Invokes the method considering the base's BeanInfo class' MethodDescriptors. Resolves method
068   * overloads by looking at the MethodDescriptors first and only then the base's class itself,
069   * choosing the most specific candidate.
070   *
071   */
072  @Override
073  public Object invoke(final ELContext context, final Object base, final Object method,
074      Class<?>[] paramTypes, final Object[] params) {
075    if (base == null || method == null) {
076      return null;
077    }
078
079    final MethodDescriptor[] methodDescriptors = getMethodDescriptors(context, base);
080
081    if (methodDescriptors != null) {
082      final Method m =
083          findMethodOrThrow(method.toString(), paramTypes, params, false, methodDescriptors);
084      paramTypes = m.getParameterTypes();
085    }
086
087    return super.invoke(context, base, method, paramTypes, params);
088  }
089
090  public Object invokeOnInterface(final ELContext context, final Class<?> baseInterface,
091      final Object base, final Object method, final Class<?>[] paramTypes, final Object[] params) {
092    if (base == null || method == null || baseInterface == null) {
093      return null;
094    }
095
096    final Method m = findMethodOrThrow(baseInterface, method.toString(), paramTypes, params, false);
097    return invoke(context, base, m, params);
098  }

100  public Object invokeIfBeanInfoOnly(final ELContext context, final Object base,
101      final Object method, final Class<?>[] paramTypes, final Object[] params) {
102    if (base == null || method == null) {
103      return null;
104    }
105
106    final Method m =
107        findMethodOrThrow(method.toString(), paramTypes, params, false,
108            getMethodDescriptors(context, base));
109    return super.invoke(context, base, method, m.getParameterTypes(), params);
110  }

124  protected Method findMethodOrThrow(final String methodName, final Class<?>[] paramTypes,
125      final Object[] params, final boolean staticOnly, final MethodDescriptor[] methodDescriptors) {
126    final Method method = findMethod(methodName, paramTypes, params, staticOnly, methodDescriptors);
127
128    if (method == null) {
129      throw new MethodNotFoundException("Method " + methodName + "for bean class "
130          + "not found or accessible");
131    }
132
133    return method;
134  }
135
136  protected Method findMethodOrThrow(final String methodName, final Class<?>[] paramTypes,
137      final Object[] params, final boolean staticOnly,
138      final Map<String, List<MethodDescriptor>> methodDescriptors) {
139    final Method method = findMethod(methodName, paramTypes, params, staticOnly, methodDescriptors);
140
141    if (method == null) {
142      throw new MethodNotFoundException("Method " + methodName + " for bean class "
143          + "not found or accessible");
144    }
145
146    return method;
147  }
148
149  protected Method findMethodOrThrow(final Class<?> klass, final String methodName,
150      final Class<?>[] paramTypes, final Object[] params, final boolean staticOnly) {
151    final Method method = findMethod(klass, methodName, paramTypes, params, staticOnly);
152
153    if (method == null) {
154      throw new MethodNotFoundException("Method " + methodName + "for class " + klass
155          + " not found or accessible");
156    }
157
158    return method;
159  }
160
161  protected Method findMethod(final Class<?> klass, String methodName, final Class<?>[] paramTypes,
162      final Object[] params, final boolean staticOnly) {
163    Method candidate = null;
164    methodName = methodName.intern();
165
166    if (paramTypes != null) {
167      try {
168        final Method method = klass.getMethod(methodName, paramTypes);
169
170        if (isMethodValid(method, staticOnly)) {
171          return method;
172        }
173      } catch (final NoSuchMethodException ex) {
174      }
175    } else {
176      for (final Method method : klass.getMethods()) {
177        if (isMethodCandidate(method, methodName, params, staticOnly, false)) {
178          if (candidate == null || suitableMethodComparator.compare(method, candidate) < 0) {
179            candidate = method;
180          }
181        }
182      }
183    }
184
185    return candidate;
186  }
187
188  protected Method findMethod(final String methodName, final Class<?>[] paramTypes,
189      final Object[] params, final boolean staticOnly, final MethodDescriptor[] methodDescriptors) {
190    final int paramCount =
191        paramTypes == null ? params == null ? 0 : params.length : paramTypes.length;
192    return findBeanInfoCandidate(paramCount, methodName.intern(), methodDescriptors, staticOnly);
193  }
194
195  protected Method findMethod(final String methodName, final Class<?>[] paramTypes,
196      final Object[] params, final boolean staticOnly,
197      final Map<String, List<MethodDescriptor>> methodDescriptors) {
198    final int paramCount =
199        paramTypes == null ? params == null ? 0 : params.length : paramTypes.length;
200    return findBeanInfoCandidate(paramCount, methodName.intern(), methodDescriptors, staticOnly);
201  }
202
203  protected <T> Method findBeanInfoCandidate(final int paramCount, final String methodName,
204      final MethodDescriptor[] methodDescriptors, final boolean staticOnly) {
205    for (final MethodDescriptor mDescriptor : methodDescriptors) {
206      if (mDescriptor.getName().equals(methodName)) {
207        final Method method = mDescriptor.getMethod();
208
209        if (isMethodValid(method, staticOnly)
210            && (method.isVarArgs() && (method.getParameterTypes().length - paramCount <= 1) 
211            || (method.getParameterTypes().length == paramCount))) {
212          return method;
213        }
214      }
215    }
216
217    return null;
218  }
219
220  protected <T> Method findBeanInfoCandidate(final int paramCount, final String methodName,
221      final Map<String, List<MethodDescriptor>> methodDescriptors, final boolean staticOnly) {
222    for (final MethodDescriptor mDescriptor : methodDescriptors.get(methodName)) {
223      if (mDescriptor.getName().equals(methodName)) {
224        final Method method = mDescriptor.getMethod();
225
226        if (isMethodValid(method, staticOnly)
227            && (method.isVarArgs() && method.getParameterTypes().length - paramCount <= 1 || method
228            .getParameterTypes().length == paramCount)) {
229          return method;
230        }
231      }
232    }
233
234    return null;
235  }

237  public static enum StaticMethodCompareStrategy {
238    STATIC_IS_MORE(true, true, false), STATIC_IS_LESS(false, true, false), STATIC_IS_MORE_IF_EQUAL(
239        true, false, true), STATIC_IS_LESS_IF_EQUAL(false, false, true), STATIC_IGNORED(false,
240            false, false);
241
242    private boolean checkBefore;
243    private boolean checkAfter;
244    private int signMore;
245
246    private StaticMethodCompareStrategy(final boolean isMore, final boolean checkBefore,
247        final boolean checkAfter) {
248      signMore = isMore ? 1 : -1;
249      this.checkBefore = checkBefore;
250      this.checkAfter = checkAfter;
251    }
252
253    public int compareBeforeParameters(final Method method1, final Method method2) {
254      return checkBefore ? signMore
255          * Boolean.compare(Modifier.isStatic(method1.getModifiers()),
256              Modifier.isStatic(method2.getModifiers())) : 0;
257    }
258
259    public int compareAfterParameters(final Method method1, final Method method2) {
260      return checkAfter ? signMore
261          * Boolean.compare(Modifier.isStatic(method1.getModifiers()),
262              Modifier.isStatic(method2.getModifiers())) : 0;
263    }
264  }
265
266  public static class SuitableMethodComparator implements Comparator<Method> {
267    private final StaticMethodCompareStrategy staticStrategy;
268
269    public SuitableMethodComparator() {
270      staticStrategy = StaticMethodCompareStrategy.STATIC_IS_MORE;
271    }
272
273    public SuitableMethodComparator(final StaticMethodCompareStrategy staticStrategy) {
274      this.staticStrategy = staticStrategy;
275    }
276
277    @Override
278    public int compare(final Method method1, final Method method2) {
279      switch (staticStrategy.compareBeforeParameters(method1, method2)) {
280        case -1:
281          return -1;
282        case 1:
283          return 1;
284      }
285
286      final Class<?> class1 = method1.getDeclaringClass();
287      final Class<?> class2 = method2.getDeclaringClass();
288
289      if (class1.equals(class2)) {
290        final Class<?>[] params1 = method1.getParameterTypes();
291        final Class<?>[] params2 = method2.getParameterTypes();
292
293        int i = 0;
294
295        for (final Class<?> pClass1 : params1) {
296          final Class<?> pClass2 = params2[i++];
297
298          if (!pClass1.equals(pClass2)) {
299            return pClass1.isAssignableFrom(pClass2) ? 1 : -1;
300          }
301        }
302
303        return staticStrategy.compareAfterParameters(method1, method2);
304      }
305
306      return class1.isAssignableFrom(class2) ? 1 : -1;
307    }
308  }

310  protected <T> boolean isMethodCandidate(final Method method, final String name, final T[] params,
311      final boolean staticOnly, final boolean isParamTypes) {
312    if (!(method.getName().equals(name) && isMethodValid(method, staticOnly))) {
313      return false;
314    }
315
316    final Class<?>[] mParamTypes = method.getParameterTypes();
317
318    if (params == null || params.length == 0) {
319      return method.isVarArgs() ? mParamTypes.length <= 1 : mParamTypes.length == 0;
320    } else if (method.isVarArgs() ? mParamTypes.length - params.length > 1
321        : mParamTypes.length != params.length) {
322      return false;
323    }
324
325    int i = 0;
326
327    for (final Class<?> pClass : mParamTypes) {
328      final Object param = params[i++];
329
330      if (isParamTypes) {
331        if (!pClass.isAssignableFrom((Class<?>) param)) {
332          return false;
333        }
334      } else {
335        if (!(param instanceof javax.el.LambdaExpression) && !pClass.isInstance(param)) {
336          return false;
337        }
338      }
339    }
340
341    return true;
342  }
343
344  private boolean isMethodValid(final Method m, final boolean staticOnly) {
345    final int mod = m.getModifiers();
346    return Modifier.isPublic(mod) && (!staticOnly || Modifier.isStatic(mod));
347  }

349  public String[] addBeanInfoSearchPath(final String[] additionalBeanInfoPath) {
350    final String[] beanInfoSearchPath = Introspector.getBeanInfoSearchPath();
351
352    final List<String> list =
353        new ArrayList<String>(beanInfoSearchPath.length + additionalBeanInfoPath.length);
354    Collections.addAll(list, beanInfoSearchPath);
355    Collections.addAll(list, additionalBeanInfoPath);
356
357    Introspector.setBeanInfoSearchPath(list.toArray(new String[0]));
358    return beanInfoSearchPath;
359  }
360
361  public BeanInfo getBeanInfo(final Object base, final String... additionalBeanInfoPath) {
362    if (base == null) {
363      return null;
364    }
365
366    BeanInfo info = null;
367    String[] beanInfoSearchPath = null;
368
369    try {
370      if (additionalBeanInfoPath != null) {
371        beanInfoSearchPath = addBeanInfoSearchPath(additionalBeanInfoPath);
372      }
373
374      info = Introspector.getBeanInfo(base instanceof Class ? (Class<?>) base : base.getClass());
375    } catch (final Exception ex) {
376    } finally {
377      if (beanInfoSearchPath != null) {
378        Introspector.setBeanInfoSearchPath(beanInfoSearchPath);
379      }
380    }
381
382    return info;
383  }
384
385  public MethodDescriptor[] getMethodDescriptors(final ELContext context, final Object base,
386      final String... additionalBeanInfoPath) {
387    final BeanInfo info = getBeanInfo(base, additionalBeanInfoPath);
388
389    if (info == null) {
390      return null;
391    }
392
393    return info.getMethodDescriptors();
394  }

396  private Object invoke(final ELContext context, final Object base, final Method method,
397      final Object[] params) {
398    if (base == null || method == null || method == null) {
399      return null;
400    }
401
402    Method baseMethod = method;
403
404    if (!method.getDeclaringClass().isAssignableFrom(base.getClass())) {
405      try {
406        baseMethod = base.getClass().getMethod(method.getName(), method.getParameterTypes());
407      } catch (final Throwable e) {
408        throw new ELException(e);
409      }
410    }
411
412    for (final Object p : params) {
413      // If the parameters is a LambdaExpression, set the ELContext
414      // for its evaluation
415      if (p instanceof javax.el.LambdaExpression) {
416        ((javax.el.LambdaExpression) p).setELContext(context);
417      }
418    }
419
420    final Object ret = ELUtil.invokeMethod(context, baseMethod, base, params);
421    context.setPropertyResolved(base, method.getName());
422    return ret;
423  }
