001package com.github.protobufel.grammar;
002
003import java.util.LinkedHashSet;
004import java.util.Set;
005
006import com.google.protobuf.Descriptors.Descriptor;
007import com.google.protobuf.Descriptors.FieldDescriptor;
008import com.google.protobuf.Descriptors.FieldDescriptor.JavaType;
009import com.google.protobuf.Descriptors.FileDescriptor;
010import com.google.protobuf.DynamicMessage;
011import com.google.protobuf.ExtensionRegistry;
012import com.google.protobuf.ExtensionRegistry.ExtensionInfo;
013
014/**
015 * Utilities for ExtensionRegistry construction.
016 * 
017 * @author protobufel@gmail.com David Tesler
018 */
019public class ExtensionRegistries {
020  private ExtensionRegistries() {}
021
022  /**
023   * Finds extension by its containing type and the dot separated multi-part name.
024   */
025  public static ExtensionInfo findExtensionByName(final ExtensionRegistry registry,
026      final Descriptor containingType, final String name) {
027    final int index = name.lastIndexOf(".");
028    final String nameLastPart = index == -1 ? name : name.substring(index + 1);
029    return registry.findImmutableExtensionByName(containingType.getFullName() + "." + nameLastPart);
030  }
031
032  /**
033   * Builds a full unmodifiable, with dependencies, ExtensionRegistry for the given FileDescriptor.
034   */
035  public static ExtensionRegistry buildFullRegistryOf(final FileDescriptor file) {
036    return addWithDependeciesToRegistry(file, ExtensionRegistry.newInstance()).getUnmodifiable();
037  }
038
039  /**
040   * Adds all extensions, including dependencies, for the given FileDescriptor to the registry.
041   */
042  public static ExtensionRegistry addWithDependeciesToRegistry(final FileDescriptor file,
043      final ExtensionRegistry registry) {
044    for (final FileDescriptor dependency : getFileWithAllDependencies(file)) {
045      addToRegistry(dependency, registry);
046    }
047
048    return registry;
049  }
050
051  private static Iterable<FileDescriptor> getFileWithAllDependencies(final FileDescriptor file) {
052    final Set<FileDescriptor> cache = new LinkedHashSet<FileDescriptor>();
053
054    cache.add(file);
055
056    // we must follow the original ProtoBuf here!
057    for (final FileDescriptor dependency : file.getDependencies()) {
058      cache.add(dependency);
059      addPublicDependencies(cache, dependency);
060    }
061
062    return cache;
063  }
064
065  private static void addPublicDependencies(final Set<FileDescriptor> cache,
066      final FileDescriptor file) {
067    for (final FileDescriptor dependency : file.getPublicDependencies()) {
068      if (cache.add(dependency)) {
069        addPublicDependencies(cache, dependency);
070      }
071    }
072  }
073
074  /**
075   * Adds all extensions, without dependencies, for the given FileDescriptor to the registry.
076   */
077  public static ExtensionRegistry addToRegistry(final FileDescriptor file,
078      final ExtensionRegistry registry) {
079    for (final FieldDescriptor extension : file.getExtensions()) {
080      addToRegistry(extension, registry);
081    }
082
083    for (final Descriptor descriptor : file.getMessageTypes()) {
084      addToRegistry(descriptor, registry);
085    }
086
087    return registry;
088  }
089
090  /**
091   * Adds all extensions for the given Descriptor to the registry.
092   */
093  public static ExtensionRegistry addToRegistry(final Descriptor descriptor,
094      final ExtensionRegistry registry) {
095    for (final FieldDescriptor extension : descriptor.getExtensions()) {
096      addToRegistry(extension, registry);
097    }
098
099    for (final Descriptor child : descriptor.getNestedTypes()) {
100      addToRegistry(child, registry);
101    }
102
103    return registry;
104  }
105
106  /**
107   * Adds extension to the registry, making a default DynamicMessage if its type is a Message.
108   */
109  public static ExtensionRegistry addToRegistry(final FieldDescriptor extension,
110      final ExtensionRegistry registry) {
111    if (extension.getJavaType() == JavaType.MESSAGE) {
112      registry.add(extension, DynamicMessage.getDefaultInstance(extension.getMessageType()));
113    } else {
114      registry.add(extension);
115    }
116
117    return registry;
118  }
119}
