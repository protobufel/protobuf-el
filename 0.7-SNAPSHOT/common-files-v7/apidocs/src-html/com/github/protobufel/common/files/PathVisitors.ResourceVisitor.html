066  public static class ResourceFileException extends RuntimeException {
067    private static final long serialVersionUID = 4984405890116628621L;
068
069    private ResourceFileException(String message, Throwable cause, boolean enableSuppression,
070        boolean writableStackTrace) {
071      super(message, cause, enableSuppression, writableStackTrace);
072    }
073
074    private ResourceFileException(String message, Throwable cause) {
075      super(message, cause);
076    }
077
078    private ResourceFileException(String message) {
079      super(message);
080    }
081
082    private ResourceFileException(Throwable cause) {
083      super(cause);
084    }
085  }

087  public static Iterable<Path> getResourceFiles(final Path rootDir, 
088      final IFileSet fileSet, final boolean followLinks, final boolean allowDuplicates, 
089      final Logger log) {
090    return getResourceFiles(rootDir, assertNonNull(Collections.singleton(fileSet)), followLinks, 
091        allowDuplicates, log);
092  }
093
094  public static Iterable<Path> getResourceFiles(final Path rootDir, 
095      final Iterable<? extends IFileSet> fileSets, final boolean followLinks, 
096      final boolean allowDuplicates, final Logger log) {
097    final List<FileSetPathMatcher<Path>> matchers = new ArrayList<>(); 
098    
099    for (IFileSet fileSet : verifyNonNull(fileSets)) {
100      @SuppressWarnings("null")
101      final @NonNull Path dir = Paths.get(verifyNonNull(fileSet).getDirectory());
102      final FileSetPathMatcher<Path> matcher = new FileSetPathMatcher<Path>(fileSet.getIncludes(),
103          fileSet.getExcludes(), dir, fileSet.isAllowDirs(), fileSet.isAllowFiles());
104      matchers.add(matcher);
105    }
106
107    final CompositePathMatcher<Path, FileSetPathMatcher<Path>> matcher = new CompositePathMatcher<>(matchers);
108    final Set<Path> result = new LinkedHashSet<>();
109
110    try {
111      final EnumSet<FileVisitOption> options = followLinks 
112          ? EnumSet.of(FileVisitOption.FOLLOW_LINKS) 
113              : EnumSet.noneOf(FileVisitOption.class);
114      final ResourceVisitor<Path> resourceVisitor = new ResourceVisitor<>(matcher, allowDuplicates, 
115          result, log, rootDir);
116      Files.walkFileTree(rootDir, options, Integer.MAX_VALUE, resourceVisitor);
117      
118      if (!resourceVisitor.getErrorMessage().isEmpty()) {
119        throw new ResourceFileException(resourceVisitor.getErrorMessage());
120      }
121      
122      @SuppressWarnings("null")
123      final @NonNull Set<Path> unmodifiableSet = Collections.unmodifiableSet(result);
124      return unmodifiableSet;
125    } catch (Exception e) {
126      log.error("fileSet caused error", e);
127      throw new ResourceFileException("fileSet caused error", e);
128    }
129  }

131  public static final class ResourceVisitor<T> extends SimpleFileVisitor<T> {
132    private static final int MATCHER_CACHE_SIZE = 1024;
133    private static boolean useEmptyCache = true; // for testing only
134    private final ContextHierarchicalMatcher<T> matcher;
135    private final boolean allowDuplicates;
136    private final Set<T> result;
137    private String errorMessage;
138    private final Logger log;
139    private boolean skipRoot = true;
140    private final IHistoryCache<Object, Object> matcherCache;
141    private final PathContext<T> pathContext;
142
143    public ResourceVisitor(final ContextHierarchicalMatcher<T> matcher, final boolean allowDuplicates, 
144        final Set<T> result, final Logger log, T rootDir) {
145      this(matcher, allowDuplicates, result, log, MATCHER_CACHE_SIZE, rootDir);
146    }
147
148    public ResourceVisitor(final ContextHierarchicalMatcher<T> matcher, final boolean allowDuplicates, 
149        final Set<T> result, final Logger log, final int maxCacheSize, final T rootDir) {
150      this.matcher = verifyNonNull(matcher);
151      this.matcherCache = getHistoryCache(matcher, maxCacheSize);
152      this.result = verifyNonNull(result);
153      this.log = verifyNonNull(log);
154      this.allowDuplicates = allowDuplicates;
155      this.errorMessage = "";
156      this.pathContext = getPathContext(verifyNonNull(rootDir));
157    }
158    
159    protected PathContext<T> getPathContext(T root) {
160      if (root instanceof Path) {
161        @SuppressWarnings("unchecked")
162        final PathContext<T> pathContext = (PathContext<T>) new SimplePathContext(
163            matcherCache.getCacheView(), (Path) root);
164        return pathContext;
165      }
166      
167      return new BasePathContext<T>(matcherCache.getCacheView(), root);
168    }
169    
170    static boolean isUseEmptyCache() {
171      // for testing only
172      return useEmptyCache;
173    }
174
175    static void setUseEmptyCache(final boolean useEmptyCache) {
176      // for testing only
177      ResourceVisitor.useEmptyCache = useEmptyCache;
178    }
179
180    private IHistoryCache<Object, Object> getHistoryCache(
181        final ContextHierarchicalMatcher<T> matcher, final int maxCacheSize) {
182      final IHistoryCache<Object, Object> matcherCache;
183      
184      if (isUseEmptyCache()) {
185        matcherCache = HistoryCaches.fakeInstance();
186      } else {
187        matcherCache = new HistoryCache<>(verifyArgument(maxCacheSize > 0, maxCacheSize));
188      }
189      
190      return matcherCache;
191    }
192
193    @Override
194    @NonNullByDefault(false)
195    public FileVisitResult preVisitDirectory(T dir, BasicFileAttributes attrs)
196        throws IOException {
197      super.preVisitDirectory(dir, attrs); // dir, attrs are checked for nonNull!
198      @SuppressWarnings("null")
199      final @NonNull T nonNullDir = dir;
200      //TODO either use or remove!
201      @SuppressWarnings({ "unused", "null" })
202      final @NonNull BasicFileAttributes nonNullAttrs = attrs;
203      
204      if (skipRoot) {
205        skipRoot = false;
206        return FileVisitResult.CONTINUE;
207      } 
208      
209      matcherCache.push();
210      final DirectoryMatchResult matcherResult = matcher.matchesDirectory(nonNullDir, pathContext);
211      
212      if (matcherResult.isMatched()) {
213        if (!result.add(nonNullDir) && !allowDuplicates) {
214          @SuppressWarnings("null")
215          @NonNull String nonNullErrorMsg = String.format("found a duplicate folder %s", nonNullDir);
216          errorMessage = nonNullErrorMsg;
217          log.error(errorMessage);
218          matcherCache.clear();
219          return FileVisitResult.TERMINATE;
220        }
221      }
222      
223      if (matcherResult.isSkip()) {
224        matcherCache.pop();
225        return FileVisitResult.SKIP_SUBTREE;
226      } else {
227        return FileVisitResult.CONTINUE;
228      }  
229    }
230
231    @Override
232    @NonNullByDefault(false)
233    public FileVisitResult postVisitDirectory(T dir, IOException exc) throws IOException {
234      super.postVisitDirectory(dir, exc);
235      matcherCache.pop();
236      return FileVisitResult.CONTINUE;
237    }
238    
239    @Override
240    @NonNullByDefault(false)
241    public FileVisitResult visitFile(T file, BasicFileAttributes attrs) throws IOException {
242      super.visitFile(file, attrs);
243      @SuppressWarnings("null")
244      final @NonNull T nonNullFile = file;
245
246      if (matcher.matches(nonNullFile, pathContext)) {
247        if (!result.add(nonNullFile) && !allowDuplicates) {
248          @SuppressWarnings("null")
249          final @NonNull String nonNullErrorMsg = String.format("found a duplicate file %s", 
250              nonNullFile);
251          errorMessage = nonNullErrorMsg;
252          log.error(errorMessage);
253          matcherCache.clear();
254          return FileVisitResult.TERMINATE;
255        }
256      }
257
258      return FileVisitResult.CONTINUE;
259    }
260
261    @Override
262    @NonNullByDefault(false)
263    public FileVisitResult visitFileFailed(T file, IOException exc) throws IOException {
264      log.error(String.format("file %s cannot be visited", verifyNonNull(file)), exc);
265      throw exc;
266    }
267    
268    public String getErrorMessage() {
269      return errorMessage;
270    }
271  }
