050package com.github.protobufel.el;

052/**
053 * Useful utility methods, taken from javax.el package in the modified and unmodified form. 
054 *  
055 * @author protobufel@gmail.com David Tesler
056 */
057public final class ELSupport {
058  private ELSupport() {}
059  
060  public static int toInteger(final Object p) {
061    if (p instanceof Integer) {
062      return ((Integer) p).intValue();
063    }
064    if (p instanceof Character) {
065      return ((Character) p).charValue();
066    }
067    if (p instanceof Number) {
068      return ((Number) p).intValue();
069    }
070    if (p instanceof String) {
071      return Integer.parseInt((String) p);
072    }
073    throw new IllegalArgumentException();
074  }
075}
