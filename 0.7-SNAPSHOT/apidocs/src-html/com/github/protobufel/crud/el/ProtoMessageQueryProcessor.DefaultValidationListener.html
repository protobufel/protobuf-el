054/**
055 * Processes queries, lists of objects, using the EL 3.0 language, including validation and error
056 * reporting.
057 * <p>
058 * Also, it allows you to specify your own JavaBeans to be used within the expression, in addition
059 * to the predefined ones:
060 * <ol>
061 * <li>{@code records} - the immutable List<Message> of the original records
062 * <li>{@code results} - the immutable List<Message> of the results being produced
063 * <li>{@code record} - the current row's Message.Builder, will be in the {@code results}
064 * <li>{@code index} - the current row's index in the results
065 * </ol>
066 * <p>
067 * You can also specify the so called empty expression to the {@link Builder} to produce the
068 * original records.
069 * <p>
070 * In addition, you can add your own {@link ValidationListener} and {@link QueryResultListener}.
071 * <p>
072 * If your expression returns {@code null}, the current result will be skipped, so this is the way
073 * to remove the original record from the results. The {@link QueryResultListener#resultAdded(int)}
074 * allows you to keep track of the being produced results. Initially, the {@code results} list is
075 * empty; and the result producing loop follows the original records. So, the originals vs the
076 * results diffs can be easily and efficiently calculated.
077 * 
078 * @author protobufel@gmail.com David Tesler
079 */
080public final class ProtoMessageQueryProcessor implements IQueryProcessor<Message> {

097  protected ProtoMessageQueryProcessor(final String expression, final String emptyExpression,
098      final Descriptor type, final QueryResultListener resultListener,
099      final ValidationListener validationListener, final IDynamicMessageProvider messageProvider) {
100    this.type = type;
101    this.resultListener = resultListener;
102    this.expression = expression;
103    this.emptyExpression = emptyExpression;
104    this.validationListener =
105        (validationListener == null) ? DEFAULT_VALIDATION_LISTENER : validationListener;
106    this.messageProvider = messageProvider;
107  }

194  public List<Message> process(final List<? extends Message> originalList, final Descriptor type,
195      final String expression, final String emptyExpression, final Map<String, Object> beans,
196      final IDynamicMessageProvider messageProvider) {
197    if (originalList == null) {
198      throw new NullPointerException();
199    } else if (originalList.isEmpty() ? (emptyExpression == null) || emptyExpression.isEmpty()
200        : (expression == null) || expression.isEmpty()) {
201      return Collections.emptyList();
202    }
203
204    final ELProcessor elp = newELProcessor();
205
206    if ((beans != null) && !beans.isEmpty()) {
207      for (final Entry<String, Object> entry : beans.entrySet()) {
208        elp.defineBean(entry.getKey(), entry.getValue());
209      }
210    }
211
212    final List<Message> results = new ArrayList<Message>();
213    final List<Message> readOnlyResults = Collections.unmodifiableList(results);
214    int index = -1;
215
216    if (originalList.isEmpty()) {
217      final Object result = elp.eval(emptyExpression);
218
219      if (!collectResult(results, type, index, result, originalList)) {
220        return Collections.emptyList();
221      }
222
223      return readOnlyResults;
224    }
225
226    final List<Message> records = Collections.unmodifiableList(originalList);
227    elp.defineBean("records", records);
228    elp.defineBean("results", readOnlyResults);
229    final ELContext context = elp.getELManager().getELContext();
230    final ValueExpression processExpr = getValueExpression(context, expression, Object.class);
231
232    for (final Message message : records) {
233      elp.defineBean("record", getBuilder(message, messageProvider));
234      elp.defineBean("index", ++index);
235      final Object result = processExpr.getValue(context);
236
237      if (!collectResult(results, type, index, result, originalList)) {
238        return Collections.emptyList();
239      }
240    }
241
242    return readOnlyResults;
243  }
244
245  private boolean collectResult(final List<Message> results, final Descriptor type,
246      final int index, final Object result, final List<? extends Message> originalList) {
247    if (result == null) {
248      // noop - record is not added to the results, i.e. skipped!
249    } else if (result instanceof Iterable) {
250      // add all non-null items, properly converted, to the results
251      for (final Object element : (Iterable<?>) result) {
252        if (!collectSingleResult(results, type, index, element, originalList)) {
253          return false;
254        }
255      }
256    } else {
257      if (!collectSingleResult(results, type, index, result, originalList)) {
258        return false;
259      }
260    }
261
262    return true;
263  }
264
265  private boolean collectSingleResult(final List<Message> results, final Descriptor type,
266      final int index, final Object result, final List<? extends Message> originalList) {
267    if (result == null) {
268      return true;
269    } else if (!(result instanceof MessageOrBuilder)) {
270      throw new IllegalArgumentException(String.format(
271          "result of original record #%s is not a MessageOrBuilder", index));
272    } else if (!validationListener.validate(type, originalList.get(index),
273        (MessageOrBuilder) result)) {
274      throw new IllegalArgumentException(String.format(
275          "result of original record #%s is of the wrong message type", index));
276    }
277
278    if (result instanceof Message.Builder) {
279      results.add(((Message.Builder) result).build());
280    } else if (result instanceof Message) {
281      results.add((Message) result);
282    }
283
284    return notifyResultAdded(index);
285  }
286
287  /**
288   * Notifies resultListener when the record added.
289   *
290   * @param index originalList record index
291   * @return true - to continue (by default), false - to abort
292   */
293  protected boolean notifyResultAdded(final int index) {
294    return (resultListener == null) ? true : resultListener.resultAdded(index);
295  }

301  protected Message.Builder getBuilder(final Message message,
302      final IDynamicMessageProvider messageProvider) {
303    if (message == null) {
304      return null;
305    } else if ((message instanceof GeneratedMessage) || (message instanceof IBuilder2)) {
306      return message.toBuilder();
307    } else {
308      return messageProvider.newBuilder(message);
309    }
310  }

312  private ValueExpression getValueExpression(final ELContext context, final String expression,
313      final Class<?> expectedType) {
314    return ELManager.getExpressionFactory().createValueExpression(context, bracket(expression),
315        expectedType);
316  }
317
318  private String bracket(final String expression) {
319    return "${" + expression + '}';
320  }

322  /**
323   * A query builder and EL 3.0 Language processor with ProtoBuf.
324   *
325   * @author protobufel@gmail.com David Tesler
326   */
327  public static final class QueryBuilder implements IQueryProcessor<Message> {
328    private Descriptor type = null;
329    private String expression = null;
330    private String emptyExpression = null;
331    private QueryResultListener resultListener = null;
332    private ValidationListener validationListener = null;
333    private Map<String, Object> beans = null;
334    private IDynamicMessageProvider messageProvider = null;
335
336    /**
337     * Constructs a default QueryBuilder
338     */
339    public QueryBuilder() {
340      messageProvider = DynamicMessage.getProvider();
341    }
342
343    /**
344     * Constructs a QueryBuilder based on another QueryBuilder.
345     * <p>
346     * All the fields of the parameter builder are copied as-is but {@code  beans}, which are not
347     * set. This is done to prevent the inadvertent original builder's {@code beans} mutation. If
348     * needed, this can be accomplished explicitly via {@link #setBeans(Map)} or
349     * {@link #addBean(String, Object)} methods.
350     *
351     * @param builder a template QueryBuilder to copy from
352     */
353    public QueryBuilder(final QueryBuilder builder) {
354      type = builder.type;
355      expression = builder.expression;
356      emptyExpression = builder.emptyExpression;
357      resultListener = builder.resultListener;
358      validationListener = builder.validationListener;
359      messageProvider = builder.messageProvider;
360      // beans = (builder.beans == null) ? null : new HashMap<String, Object>(builder.beans);
361    }
362
363    public QueryBuilder(final ProtoMessageQueryProcessor processor) {
364      type = processor.type;
365      expression = processor.expression;
366      emptyExpression = processor.emptyExpression;
367      resultListener = processor.resultListener;
368      validationListener = processor.validationListener;
369      messageProvider = processor.messageProvider;
370    }
371

373      return messageProvider;
374    }

376    public QueryBuilder setMessageProvider(final IDynamicMessageProvider messageProvider) {
377      this.messageProvider = messageProvider;
378      return this;
379    }
380
381    public Map<String, Object> getBeans() {
382      return beans;
383    }
384
385    /**
386     * Sets this QueryBuilder's beans to the shallow copy of the parameter.
387     *
388     * @param beans a map of a bean name and its value to be copied over
389     */
390    public QueryBuilder setBeans(final Map<String, Object> beans) {
391      this.beans = beans == null ? null : new HashMap<String, Object>(beans);
392      return this;
393    }
394
395    public QueryBuilder addBean(final String name, final Object bean) {
396      if (beans == null) {
397        beans = new HashMap<String, Object>(beans);
398      }
399
400      beans.put(name, bean);
401      return this;
402    }

449    public ProtoMessageQueryProcessor build() {
450      if (messageProvider == null) {
451        throw new NullPointerException();
452      }

455          new ProtoMessageQueryProcessor(expression, emptyExpression, type, resultListener,
456              validationListener, messageProvider);

473  public static class DefaultValidationListener implements ValidationListener {
474
475    @Override
476    public boolean validate(final Descriptor type, final Message original,
477        final MessageOrBuilder result) {
478      if (type == null) {
479        return original == null ? true : original.getDescriptorForType().equals(
480            result.getDescriptorForType());
481      } else {
482        return type.equals(result.getDescriptorForType());
483      }
484    }
