028package com.github.protobufel.grammar;
029
030import java.util.BitSet;
031import java.util.Iterator;
032
033import org.antlr.v4.runtime.ANTLRErrorListener;
034import org.antlr.v4.runtime.BaseErrorListener;
035import org.antlr.v4.runtime.Parser;
036import org.antlr.v4.runtime.RecognitionException;
037import org.antlr.v4.runtime.Recognizer;
038import org.antlr.v4.runtime.atn.ATNConfigSet;
039import org.antlr.v4.runtime.dfa.DFA;
040import org.slf4j.Logger;
041import org.slf4j.LoggerFactory;

043/**
044 * Error handlers extending {@link ANTLRErrorListener} with {@code validationError}s.
045 * 
046 * @author protobufel@gmail.com David Tesler
047 */
048public final class ErrorListeners {
049  private ErrorListeners() {}
050
051  /**
052   * A base interface for all ErrorListeners reporting syntax and validation errors.
053   * 
054   * @author protobufel@gmail.com David Tesler
055   */
056  public interface IProtoErrorListener extends ANTLRErrorListener {
057    @Override
058    public void syntaxError(Recognizer<?, ?> recognizer, Object offendingSymbol, int line,
059        int charPositionInLine, String msg, RecognitionException e);
060
061    /**
062     * Reports a validation error. All parameters are nullable.
063     */
064    public void validationError(Integer line, Integer charPositionInLine, String msg,
065        RuntimeException exception);
066
067    /**
068     * Returns the proto name this listener is reporting about.
069     */
070    public String getProtoName();
071  }
072
073  /**
074   * Extends {@link IProtoErrorListener} with the ability to set {@code protoName}.
075   *
076   * @author protobufel@gmail.com David Tesler
077   */
078  public interface IBaseProtoErrorListener extends IProtoErrorListener {
079    public void setProtoName(String protoName);
080  }
081
082  /**
083   * A {@link BaseErrorListener} with the settable {@code protoName}.
084   *
085   * @author protobufel@gmail.com David Tesler
086   */
087  public static class BaseProtoErrorListener extends BaseErrorListener implements
088      IBaseProtoErrorListener {
089    private String protoName;
090
091    public BaseProtoErrorListener() {
092      protoName = null;
093    }
094
095    public BaseProtoErrorListener(final String protoName) {
096      this.protoName = protoName;
097    }
098
099    @Override
100    public String getProtoName() {
101      return protoName;
102    }
103
104    @Override
105    public void setProtoName(final String protoName) {
106      if (protoName == null) {
107        throw new NullPointerException();
108      }
109
110      this.protoName = protoName;
111    }
112
113    @Override
114    public void syntaxError(final Recognizer<?, ?> recognizer, final Object offendingSymbol,
115        final int line, final int charPositionInLine, final String msg, final RecognitionException e) {}
116
117    @Override
118    public void validationError(final Integer line, final Integer charPositionInLine,
119        final String msg, final RuntimeException exception) {}
120  }
121
122  /**
123   * A composite {@link BaseErrorListener} delegating to its list of delegate error listeners.
124   * 
125   * @author protobufel@gmail.com David Tesler
126   */
127  public static final class ProxyProtoErrorListener extends BaseProtoErrorListener {
128    private final Iterable<? extends IBaseProtoErrorListener> delegates;
129
130    public ProxyProtoErrorListener(final Iterable<? extends IBaseProtoErrorListener> delegates) {
131      if (delegates == null) {
132        throw new NullPointerException("delegates");
133      }
134
135      this.delegates = delegates;
136    }
137
138    @Override
139    public String getProtoName() {
140      final Iterator<? extends IBaseProtoErrorListener> iterator = delegates.iterator();
141      return iterator.hasNext() ? iterator.next().getProtoName() : null;
142    }
143
144    @Override
145    public void setProtoName(final String protoName) {
146      for (final IBaseProtoErrorListener listener : delegates) {
147        listener.setProtoName(protoName);
148      }
149    }
150
151    @Override
152    public void reportAmbiguity(final Parser recognizer, final DFA dfa, final int startIndex,
153        final int stopIndex, final boolean exact, final BitSet ambigAlts, final ATNConfigSet configs) {
154      for (final IBaseProtoErrorListener listener : delegates) {
155        listener.reportAmbiguity(recognizer, dfa, startIndex, stopIndex, exact, ambigAlts, configs);
156      }
157    }
158
159    @Override
160    public void reportAttemptingFullContext(final Parser recognizer, final DFA dfa,
161        final int startIndex, final int stopIndex, final BitSet conflictingAlts,
162        final ATNConfigSet configs) {
163      for (final IBaseProtoErrorListener listener : delegates) {
164        listener.reportAttemptingFullContext(recognizer, dfa, startIndex, stopIndex,
165            conflictingAlts, configs);;
166      }
167    }
168
169    @Override
170    public void reportContextSensitivity(final Parser recognizer, final DFA dfa,
171        final int startIndex, final int stopIndex, final int prediction, final ATNConfigSet configs) {
172      for (final IBaseProtoErrorListener listener : delegates) {
173        listener.reportContextSensitivity(recognizer, dfa, startIndex, stopIndex, prediction,
174            configs);;
175      }
176    }
177
178    @Override
179    public void syntaxError(final Recognizer<?, ?> recognizer, final Object offendingSymbol,
180        final int line, final int charPositionInLine, final String msg, final RecognitionException e) {
181      for (final IBaseProtoErrorListener listener : delegates) {
182        listener.syntaxError(recognizer, offendingSymbol, line, charPositionInLine, msg, e);;
183      }
184    }
185
186
187    @Override
188    public void validationError(final Integer line, final Integer charPositionInLine,
189        final String msg, final RuntimeException exception) {
190      for (final IBaseProtoErrorListener listener : delegates) {
191        listener.validationError(line, charPositionInLine, msg, exception);
192      }
193    }
194  }
195
196  /**
197   * An error listener printing to {@link System.err}.
198   * 
199   * @author protobufel@gmail.com David Tesler
200   */
201  public static class ConsoleProtoErrorListener extends BaseProtoErrorListener {
202
203    public ConsoleProtoErrorListener() {
204      super();
205    }
206
207    public ConsoleProtoErrorListener(final String protoName) {
208      super(protoName);
209    }
210
211    @Override
212    public void syntaxError(final Recognizer<?, ?> recognizer, final Object offendingSymbol,
213        final int line, final int charPositionInLine, final String msg, final RecognitionException e) {
214      printError(line, charPositionInLine, msg);
215    }
216
217    @Override
218    public void validationError(final Integer line, final Integer charPositionInLine,
219        final String msg, final RuntimeException exception) {
220      printError(line, charPositionInLine, msg);
221    }
222
223    protected void printError(final int line, final int charPositionInLine, final String msg) {
224      System.err.println("proto '" + getProtoName() + "' @ line " + line + ":" + charPositionInLine
225          + " " + msg);
226    }
227  }
228
229  /**
230   * A logging error listener using org.slf4j.
231   *
232   * @author protobufel@gmail.com David Tesler
233   */
234  public static class LogProtoErrorListener extends BaseProtoErrorListener {
235    private static final String VALIDATION_ERROR_LOG_MSG_FORMAT =
236        "validationError proto '{}' @{}:{} msg:'{}' exception:'{}'";
237    private static final String SYNTAX_ERROR_LOG_MSG_FORMAT =
238        "syntaxError proto '{}' @{}:{} msg:'{}' recognizer:'{}' offendingSymbol:'{}' exception:'{}'";
239    private Logger logger;
240    private final IBaseProtoErrorListener delegate;
241
242    public LogProtoErrorListener() {
243      this(null, new BaseProtoErrorListener());
244    }
245
246    public LogProtoErrorListener(final String protoName) {
247      this(protoName, new BaseProtoErrorListener(protoName));
248    }
249
250    public LogProtoErrorListener(final IBaseProtoErrorListener delegate) {
251      this(null, delegate);
252    }
253
254    public LogProtoErrorListener(final String protoName, final IBaseProtoErrorListener delegate) {
255      super(protoName);
256      logger = LoggerFactory.getLogger(getClass());
257      this.delegate = delegate;
258    }
259
260    @Override
261    public void setProtoName(final String protoName) {
262      super.setProtoName(protoName);
263      delegate.setProtoName(protoName);
264    }
265
266    public LogProtoErrorListener setLogger(final Logger logger) {
267      this.logger = logger;
268      return this;
269    }
270
271    public LogProtoErrorListener setLogger(final Class<?> classToLog) {
272      logger = LoggerFactory.getLogger(classToLog);
273      return this;
274    }
275
276    @Override
277    public void syntaxError(final Recognizer<?, ?> recognizer, final Object offendingSymbol,
278        final int line, final int charPositionInLine, final String msg, final RecognitionException e) {
279      logger.error(SYNTAX_ERROR_LOG_MSG_FORMAT, getProtoName(), line, charPositionInLine, msg,
280          recognizer, offendingSymbol, e);
281      delegate.syntaxError(recognizer, offendingSymbol, line, charPositionInLine, msg, e);
282    }
283
284    @Override
285    public void validationError(final Integer line, final Integer charPositionInLine,
286        final String msg, final RuntimeException exception) {
287      logValidationError(line, charPositionInLine, msg, exception);
288      delegate.validationError(line, charPositionInLine, msg, exception);
289    }
290
291    protected void logValidationError(final Integer line, final Integer charPositionInLine,
292        final String msg, final RuntimeException exception) {
293      logger.error(VALIDATION_ERROR_LOG_MSG_FORMAT, getProtoName(), line, charPositionInLine, msg,
294          exception);
295    }
296  }
