290    private Pattern convertGlobToRegex(final String globPattern) {
291      return Pattern.compile(globPattern.replaceAll("\\\\", "/").replaceAll("[.]", "[.]")
292          .replaceAll("[?]", ".").replaceAll("[*]{2}+", ".*?")
293          .replaceAll("(?<![.])[*]{1}+(?![?])", "[^/]*"));
294    }

296    /**
297     * Adds source .proto files, by regex pattern, for processing.
298     * 
299     * @param baseDir a base directory to get .proto files from
300     * @param pattern a file regex pattern, with canonical {@code /} separator
301     */
302    public Builder addFilesByRegex(final File rootDir, final Pattern pattern) {
303      final List<String> targetFiles = new ArrayList<String>();
304      // final File dir = new File(rootDir.toURI().normalize());
305      final File dir = getCanonicalFile(rootDir);
306      processChildren(dir, dir.toURI(), targetFiles, pattern);
307      return addFiles(dir, targetFiles.toArray(new String[0]));
308    }
309
310    private void processChildren(final File dir, final URI rootURI, final List<String> targetFiles,
311        final Pattern pattern) {
312      for (final File child : dir.listFiles()) {
313        final String file = rootURI.relativize(child.toURI()).getPath();
314        final Matcher matcher = pattern.matcher(file);
315
316        if (child.isDirectory()) {
317          if (!matcher.matches() && matcher.hitEnd()) {
318            processChildren(child, rootURI, targetFiles, pattern);
319          }
320        } else {
321          if (matcher.matches()) {
322            targetFiles.add(file);
323          }
324        }
325      }
326    }

328    /**
329     * Adds source .proto files, based on the rootDir.
330     * 
331     * @param rootDir a base directory to get .proto files from
332     * @param protoFiles file paths relative to the rootDir
333     */
334    public Builder addFiles(final File rootDir, final String... protoFiles) {
335      if (rootDir == null || protoFiles == null || protoFiles.length == 0) {
336        throw new NullPointerException();
337      }
338
339      // final URI dir = baseDir.toURI().normalize();
340      final URI dirURI = getCanonicalFile(rootDir).toURI();
341      Set<URI> fileSet = dirs.get(dirURI);
342
343      if (fileSet == null) {
344        fileSet = new LinkedHashSet<URI>();
345        dirs.put(dirURI, fileSet);
346      }
347
348      for (final String file : protoFiles) {
349        fileSet.add(dirURI.relativize(URI.create(file)).normalize());
350      }
351
352      return this;
353    }

355    private File getCanonicalFile(final File file) {
356      try {
357        return file.getCanonicalFile();
358      } catch (final IOException e) {
359        throw new RuntimeException(e);
360      }
361    }

363    /**
364     * Adds .proto resources, based on the rootURI. Imports will be treated as URIs.
365     * 
366     * @param rootURI a base URI to get .proto resources from
367     * @param uris proto URIs relative to the rootURI
368     */
369    public Builder addResources(final URI rootURI, final String... uris) {
370      if (rootURI == null || !rootURI.isAbsolute() || uris == null || uris.length == 0) {
371        throw new NullPointerException();
372      }
373
374      final URI dir = rootURI.normalize();
375      Set<URI> fileSet = dirs.get(dir);
376
377      if (fileSet == null) {
378        fileSet = new LinkedHashSet<URI>();
379        dirs.put(dir, fileSet);
380      }
381
382      for (final String uri : uris) {
383        fileSet.add(URI.create(uri).normalize());
384      }
385
386      return this;
387    }

389    private ParsedContext parseString(final String source, final String protoName) {
390      final ANTLRInputStream fis = new ANTLRInputStream(source);
391      return parseProto(fis, protoName);
392    }

394    private ParsedContext parseFile(final URI rootDir, final URI proto) {
395      Throwable exception = null;
396      InputStream is = null;
397
398      try {
399        is = rootDir.resolve(proto).toURL().openStream();
400        final ANTLRInputStream ais = new ANTLRInputStream(is);
401        return parseProto(ais, proto.toString());
402      } catch (final IOException e) {
403        exception = e;
404      } finally {
405        if (is != null) {
406          try {
407            is.close();
408          } catch (final IOException e) {
409            if (exception == null) {
410              throw new RuntimeException(e);
411            }
412          }
413        }
414
415        if (exception != null) {
416          // FIXME report error instead
417          throw new RuntimeException(exception);
418        }
419      }
420
421      return null;
422    }

424    @SuppressWarnings("unused")
425    private ParsedContext parseFile(final File rootDir, final String protoName) {
426      Throwable exception = null;
427
428      try {
429        final File file = new File(rootDir, protoName);
430        final ANTLRFileStream ais = new ANTLRFileStream(file.getCanonicalPath());
431        return parseProto(ais, protoName);
432      } catch (final IOException e) {
433        exception = e;
434      } finally {
435        if (exception != null) {
436          throw new RuntimeException(exception);
437        }
438      }
439
440      return null;
441    }

443    private ParsedContext parseProto(final ANTLRInputStream is, final String protoName) {
444      final ProtoLexer lexer = new ProtoLexer(is);
445      final CommonTokenStreamEx tokens = new CommonTokenStreamEx(lexer);
446      final ProtoParser parser = new ProtoParser(tokens);
447      parser.setErrorHandler(errorStrategy);
448      parser.removeErrorListeners();
449      errorListener.setProtoName(protoName);
450      parser.addErrorListener(errorListener);
451      // parser.notifyErrorListeners("!!!BEGIN!!!!");
452      parser.setBuildParseTree(true); // tell ANTLR to build a parse tree
453      final ProtoContext tree = parser.proto();
454      final ProtoFileParser protoParser = new ProtoFileParser(protoName, errorListener);
455      final ParseTreeWalker walker = new ParseTreeWalker();
456      walker.walk(protoParser, tree);
457      return protoParser.getParsed();
458    }

479     * Builds a mutable list of canonical {@link FileDescriptorProto}s like {@link #buildProtos()},
480     * but also allows to suppress the reserved space validation; useful for testing.
481     */
482    List<FileDescriptorProto> buildProtos(final boolean validateNotInReservedSpace) {
483      final Collection<FileDescriptor> files = build(validateNotInReservedSpace).values();
484      final List<FileDescriptorProto> protoList = new ArrayList<FileDescriptorProto>(files.size());
485
486      for (final FileDescriptor file : files) {
487        protoList.add(file.toProto());
488      }
489
490      return protoList;
491    }

494     * Builds a FileDescriptor map, sorted in the original order, with regard to the reserved space.
495     * Any source proto in the {@code google.protobuf} reserved space will cause a validation error.
496     * The source {@code descriptor.proto} with the {@code google.protobuf} package, if present,

499     *

504      return build(true);

507    /**
508     * Builds a sorted FileDescriptor map, in the original order. Useful for testing within the
509     * reserved space; the {@link #build()} delegates to this method with
510     * {@code validateNotInReservedSpace = true}.
511     *
512     * @param validateNotInReservedSpace if true, disallows protos in reseved space, which is the
513     *        {@code "google.protobuf"}
514     * @return a sorted FileDescriptor map based on the source protos
515     */
516    Map<String, FileDescriptor> build(final boolean validateNotInReservedSpace) {
517      final List<ParsedContext> protos = new ArrayList<ParsedContext>();
518
519      if (!sourceProtos.isEmpty()) {
520        for (final FileDescriptorProto proto : sourceProtos) {
521          protos.add(new ParsedContext(proto));
522        }
523      }
524
525      if (!dirs.isEmpty()) {
526        for (final Entry<URI, Set<URI>> entry : dirs.entrySet()) {
527          final URI rootDir = entry.getKey();
528
529          for (final URI proto : entry.getValue()) {
530            protos.add(parseFile(rootDir, proto));
531          }
532        }
533      }
534
535      if (!sources.isEmpty()) {
536        for (final Entry<URI, String> entry : sources.entrySet()) {
537          final String protoSource = entry.getValue();
538          protos.add(parseString(protoSource, entry.getKey().toString()));
539        }
540      }
541
542      return buildFiles(protos, validateNotInReservedSpace);
543    }

545    private Map<String, FileDescriptor> buildFiles(final List<ParsedContext> protos,
546        final boolean validateNotInReservedSpace) {
547      String protoName = null;
548
549      try {
550        final Map<String, FileDescriptor> cache =
551            buildFilesFrom(protos, errorListener, showCircularTwice, showExistingNonCircular,
552                validateNotInReservedSpace);
553        // FIXME get protoSet from BuildFrom!
554        final Map<String, FileDescriptor> resultCache = new LinkedHashMap<String, FileDescriptor>();
555
556        for (final ParsedContext parsedContext : protos) {
557          // there can be some bad protos, so check for null first!
558          protoName = parsedContext.getProto().getName();
559          final FileDescriptor fileDescriptor = cache.get(protoName);
560
561          if (fileDescriptor == null) {
562            errorListener.validationError(null, null,
563                String.format("the proto '%s' is not in the file cache!", protoName), null);
564          } else {
565            // add the cached fileDescriptor in the original order!
566            resultCache.put(fileDescriptor.getName(), fileDescriptor);
567          }
568        }
569
570        return resultCache;
571      } catch (final RuntimeException e) {
572        errorListener.validationError(null, null, "protoName=" + protoName, e);
573      }
574
575      return Collections.emptyMap();
576    }

578    private Map<String, FileDescriptor> buildFilesFrom(
579    /* FIXME change to Map? */final List<ParsedContext> protos,
580        final IBaseProtoErrorListener errorListener, final boolean showCircularTwice,
581        final boolean showExistingNonCircular, final boolean validateNotInReservedSpace) {
582      final Map<String, FileDescriptor> fileMap = new LinkedHashMap<String, FileDescriptor>();
583      final LinkedList<ParsedContext> fileProtos = new LinkedList<ParsedContext>();
584      final FileDescriptor[] emptyDependencies = new FileDescriptor[0];
585      final Set<String> protoNames = new HashSet<String>();
586      boolean addedDescriptorProto = true;
587      boolean validateNotInReservedSpaceError = false;
588      boolean foundDuplicateProtos = false;
589
590      // it is crucial to replace or add the default runtime DescriptorProtos.getDescriptor(),
591      // otherwise all message options will fail!!!
592      for (final ListIterator<ParsedContext> iterator = protos.listIterator(); iterator.hasNext();) {
593        final ParsedContext parsedContext = iterator.next();
594        final String protoName = parsedContext.getProto().getName();
595
596        if (!protoNames.add(protoName)) {
597          foundDuplicateProtos = true;
598          errorListener.validationError(null, null,
599              String.format("a duplicate proto %s", protoName), null);
600        }
601
602        if (foundDuplicateProtos) {
603          continue;
604        }
605
606        if (GOOGLE_PROTOBUF_DESCRIPTOR_PROTO.equals(protoName)) {
607          addedDescriptorProto = false;
608          fileMap.put(GOOGLE_PROTOBUF_DESCRIPTOR_PROTO, DescriptorProtos.getDescriptor());
609          iterator.set(ParsedContext.getDescriptorProtoParsedContext());
610        } else {
611          if (validateNotInReservedSpace
612              && !validateNotInReservedSpace(parsedContext.getProto(), errorListener)) {
613            validateNotInReservedSpaceError = true;
614          }
615        }
616      }
617
618      if (validateNotInReservedSpaceError || foundDuplicateProtos) {
619        return Collections.emptyMap();
620      }
621
622      if (addedDescriptorProto) {
623        fileMap.put(GOOGLE_PROTOBUF_DESCRIPTOR_PROTO, DescriptorProtos.getDescriptor());
624        protoNames.add(GOOGLE_PROTOBUF_DESCRIPTOR_PROTO);
625      }
626
627      try {
628        // find all leaves
629        for (final ParsedContext fileProto : protos) {
630          // FIXME if public dependency don't build this proto at all and replace by the referred?
631          if (GOOGLE_PROTOBUF_DESCRIPTOR_PROTO.equals(fileProto.getProto().getName())) {
632            continue;
633          } else if (fileProto.getProto().getDependencyCount() == 0) {
634            buildAndCacheProto(fileMap, fileProto, emptyDependencies, errorListener);
635          } else if (!protoNames.containsAll(fileProto.getProto().getDependencyList())) {
636            for (final String dependency : fileProto.getProto().getDependencyList()) {
637              if (!protoNames.contains(dependency)) {
638                errorListener.validationError(null, null, "has a non-existent dependency on '"
639                    + dependency + "'", null);
640              }
641            }
642          } else {
643            // FIXME do processing after this validation loop!
644            fileProtos.addFirst(fileProto);
645          }
646        }
647
648        while (!fileProtos.isEmpty()) {
649          // find all fileProto(-s) with only dependencies from the fileMap and
650          // make them into files
651          boolean isLeafFound = false;
652
653          for (final Iterator<ParsedContext> iterator = fileProtos.iterator(); iterator.hasNext();) {
654            final ParsedContext fileProto = iterator.next();
655            final List<String> dependencyList = fileProto.getProto().getDependencyList();
656            FileDescriptor[] dependencies;
657
658            if (dependencyList.isEmpty()) {
659              dependencies = emptyDependencies;
660            } else if (!fileMap.keySet().containsAll(fileProto.getProto().getDependencyList())) {
661              continue;
662            } else {
663              dependencies = new FileDescriptor[dependencyList.size()];
664              int i = 0;
665
666              for (final String fileName : dependencyList) {
667                dependencies[i++] = fileMap.get(fileName);
668              }
669            }
670
671            buildAndCacheProto(fileMap, fileProto, dependencies, errorListener);
672            iterator.remove();
673            isLeafFound = true;
674          }
675
676          if (!isLeafFound) {
677            final Map<String, Set<String>> multimap = new LinkedHashMap<String, Set<String>>();
678
679            for (final ParsedContext fileProto : fileProtos) {
680              final LinkedHashSet<String> dependencySet =
681                  new LinkedHashSet<String>(fileProto.getProto().getDependencyList());
682              dependencySet.removeAll(fileMap.keySet());
683              multimap.put(fileProto.getProto().getName(), dependencySet);
684            }
685
686            for (final Entry<String, Set<String>> entry : multimap.entrySet()) {
687              final String protoName = entry.getKey();
688              errorListener.setProtoName(protoName);
689
690              for (final Iterator<String> iterator = entry.getValue().iterator(); iterator
691                  .hasNext();) {
692                final String dependency = iterator.next();
693
694                if (!multimap.keySet().contains(dependency)) {
695                  // a non-existent dependency
696                  errorListener.validationError(null, null, "has a non-existent dependency on '"
697                      + dependency + "'", null);
698                } else if (showCircularTwice ? multimap.get(dependency).contains(protoName)
699                    : multimap.get(dependency).remove(protoName)) {
700                  // a circular dependency
701                  errorListener.validationError(null, null, "has a circular dependency on '"
702                      + dependency + "'", null);
703                } else if (showExistingNonCircular) {
704                  // a dependency on existing but not resolvable
705                  errorListener.validationError(null, null,
706                      "has a dependency on existing but not resolvable '" + dependency + "'", null);
707                }
708              }
709            }
710
711            // FIXME or should we instead return Collections.emptyMap();
712            // for now, returns all valid FileDescriptors, in order of leafs to roots
713            return fileMap;
714          }
715        }
716      } finally {
717        if (addedDescriptorProto) {
718          // get rid of our descriptor.proto before we return!
719          fileMap.remove(GOOGLE_PROTOBUF_DESCRIPTOR_PROTO);
720        }
721      }
722
723      return fileMap;
724    }

727        final ParsedContext fileProto, final FileDescriptor[] dependencies,

736            fileBuilder.setProto(fileProto.getProto()).addDependencies(dependencies).build();

748    private boolean validateNotInReservedSpace(final FileDescriptorProtoOrBuilder proto,
749        final IBaseProtoErrorListener errorListener) {
750      if (proto.hasPackage()
751          && (proto.getPackage().equals("google.protobuf") || proto.getPackage().startsWith(
752              "google.protobuf."))) {
753        errorListener.validationError(null, null,
754            String.format("proto %s has package in reserved space", proto.getName()), null);
755        return false;
756      } else {
757        return true;
758      }
759    }
