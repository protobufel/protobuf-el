028package com.github.protobufel.el;
029
030import static com.github.protobufel.el.ProtoLists.getBuilderForRepeatedField;
031import static com.github.protobufel.el.ProtoLists.getFieldClass;
032import static com.github.protobufel.el.ProtoLists.verifyAndConvertField;
033
034import java.beans.FeatureDescriptor;
035import java.util.ArrayList;
036import java.util.Iterator;
037import java.util.List;
038
039import javax.el.ELContext;
040import javax.el.ELResolver;
041import javax.el.MethodNotFoundException;
042import javax.el.PropertyNotFoundException;
043import javax.el.PropertyNotWritableException;
044
045import com.google.protobuf.Descriptors.Descriptor;
046import com.google.protobuf.Descriptors.FieldDescriptor;
047import com.google.protobuf.Descriptors.FieldDescriptor.JavaType;
048import com.google.protobuf.Message;
049import com.google.protobuf.MessageOrBuilder;
050
051/**
052 * ProtoBuf Builder ELResolver.
053 *
054 * @see ELResolver
055 * @author protobufel@gmail.com David Tesler
056 */
057public class BuilderELResolver extends ELResolver {
058  private final boolean isReadOnly;
059  private final boolean isStrictMode;
060
061  public BuilderELResolver() {
062    this(false, false);
063  }
064
065  public BuilderELResolver(final boolean isReadOnly) {
066    this(isReadOnly, false);
067  }
068
069  public BuilderELResolver(final boolean isReadOnly, final boolean isStrictMode) {
070    this.isReadOnly = isReadOnly;
071    this.isStrictMode = isStrictMode;
072  }
073
074  protected boolean resolveType(final Object base) {
075    return base instanceof Message.Builder;
076  }
077
078  @Override
079  public Iterator<FeatureDescriptor> getFeatureDescriptors(final ELContext context,
080      final Object base) {
081    if (resolveType(base)) {
082      final List<FieldDescriptor> fields =
083          ((MessageOrBuilder) base).getDescriptorForType().getFields();
084      final List<FeatureDescriptor> list = new ArrayList<FeatureDescriptor>(fields.size());
085
086      for (final FieldDescriptor field : fields) {
087        final FeatureDescriptor descriptor = new FeatureDescriptor();
088        descriptor.setName(field.getName());
089        descriptor.setDisplayName(field.getName());
090        descriptor.setShortDescription(field.getFullName());
091        descriptor.setExpert(false);
092        descriptor.setHidden(false);
093        descriptor.setPreferred(true);
094        descriptor.setValue(TYPE, getFieldClass(field, true));
095        descriptor.setValue(RESOLVABLE_AT_DESIGN_TIME, Boolean.TRUE);
096        list.add(descriptor);
097      }
098
099      return list.iterator();
100    }
101
102    return null;
103  }
104
105  @Override
106  public Class<?> getType(final ELContext context, final Object base, final Object property) {
107    if (context == null) {
108      throw new NullPointerException();
109    }
110
111    if (property != null && resolveType(base)) {
112      context.setPropertyResolved(true);
113      return getFieldClass(getPropertyFieldDescriptor((MessageOrBuilder) base, property), true);
114    }
115
116    return null;
117  }
118
119  protected FieldDescriptor getPropertyFieldDescriptor(final MessageOrBuilder mob,
120      final Object property) {
121    final Descriptor descriptor = mob.getDescriptorForType();
122    FieldDescriptor field = null;
123
124    if (property instanceof String) {
125      field = descriptor.findFieldByName(property.toString());
126    } else if (property instanceof FieldDescriptor) {
127      field = FieldDescriptor.class.cast(property);
128
129      if (field.getContainingType() != descriptor) {
130        field = null;
131      }
132    }
133
134    if (field == null) {
135      throw new PropertyNotFoundException();
136    }
137
138    return field;
139  }
140
141  @Override
142  public Class<?> getCommonPropertyType(final ELContext context, final Object base) {
143    if (context == null) {
144      throw new NullPointerException();
145    }
146
147    if (resolveType(base)) {
148      context.setPropertyResolved(true);
149      return Object.class;
150    }
151
152    return null;
153  }
154
155  @Override
156  public Object getValue(final ELContext context, final Object base, final Object property) {
157    if (context == null) {
158      throw new NullPointerException();
159    }
160
161    if (property != null && resolveType(base)) {
162      context.setPropertyResolved(base, property);
163      final Message.Builder builder = (Message.Builder) base;
164      final FieldDescriptor field = getPropertyFieldDescriptor(builder, property);
165
166      if (field.isRepeated()) {
167        return getBuilderForRepeatedField(builder, field);
168      } else {
169        if (field.getJavaType() == JavaType.MESSAGE) {
170          return builder.getFieldBuilder(field);
171        } else {
172          return builder.getField(field);
173        }
174      }
175    }
176
177    return null;
178  }
179
180  @Override
181  public void setValue(final ELContext context, final Object base, final Object property,
182      final Object val) {
183    if (context == null) {
184      throw new NullPointerException();
185    }
186
187    if (property != null && resolveType(base)) {
188      context.setPropertyResolved(base, property);
189
190      if (isReadOnly(context, base, property)) {
191        throw new PropertyNotWritableException();
192      }
193
194      if (val == null) {
195        throw new IllegalArgumentException(String.format("property %s cannot be set to null",
196            property));
197      }
198
199      try {
200        final Message.Builder builder = (Message.Builder) base;
201        final FieldDescriptor field = getPropertyFieldDescriptor(builder, property);
202        builder.setField(field, verifyAndConvertField(val, field));
203      } catch (final UnsupportedOperationException ex) {
204        throw new PropertyNotWritableException();
205      }
206    }
207  }
208
209  @Override
210  public boolean isReadOnly(final ELContext context, final Object base, final Object property) {
211    if (context == null) {
212      throw new NullPointerException();
213    }
214
215    if (property != null && resolveType(base)) {
216      context.setPropertyResolved(true);
217      return isReadOnly;
218    }
219
220    return false;
221  }
222
223  @Override
224  public Object invoke(final ELContext context, final Object base, final Object method,
225      final Class<?>[] paramTypes, final Object[] params) {
226    if (resolveType(base)) {
227      if (isStrictMode) {
228        throw new MethodNotFoundException();
229      }
230    }
231
232    return null;
233  }
234}
