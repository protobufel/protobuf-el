056  public static final class ExtendedPattern implements IPatternSupport {
057    private final List<Pattern> includes;
058    private final List<Pattern> excludes;

060    @SuppressWarnings("null")
061    public ExtendedPattern(final Collection<Pattern> includes, final Collection<Pattern> excludes, 
062        final int flags) {
063      @NonNull Collection<Pattern> nonNullIncludes = verifyNonNullIterable(includes);
064      this.includes = Collections.unmodifiableList(new ArrayList<Pattern>(
065          verifyArgument(!nonNullIncludes.isEmpty(), nonNullIncludes, "includes must be non-empty")));
066      this.excludes = Collections.unmodifiableList(new ArrayList<Pattern>(verifyNonNullIterable(excludes)));
067    }
068
069    private ExtendedPattern(final Iterable<String> includes, final Iterable<String> excludes, 
070        final int flags) {
071      this.includes = getCompiledPatterns(includes, flags);
072      this.excludes = getCompiledPatterns(excludes, flags);
073    }
074    
075    private List<Pattern> getCompiledPatterns(final Iterable<String> regexes, final int flags) {
076      final @NonNull ArrayList<Pattern> patterns = new ArrayList<Pattern>();
077      
078      for (String regex : regexes) {
079        patterns.add(Pattern.compile(verifyNonNull(regex), flags));
080      }
081      
082      @SuppressWarnings("null")
083      @NonNull List<Pattern> unmodifiableList = Collections.unmodifiableList(patterns);
084      return unmodifiableList;
085    }

344    public boolean matches() {
345      lastOp = MatcherOp.MATCHES;
346      lastStart = -1;
347      int i = -1;
348      
349      for (Pattern pattern : getExcludes()) {
350        i++;
351        
352        if (delegate.usePattern(pattern).matches()) {
353          includeIndex = -1;
354          return false;
355        }
356      }
357      
358      i= -1;
359      
360      for (Pattern pattern : getIncludes()) {
361        i++;
362        
363        if (delegate.usePattern(pattern).matches()) {
364          includeIndex = i;
365          return true;
366        }
367      }
368      
369      includeIndex = -1;
370      return false;
371    }
372    
373    @Override
374    public boolean find() {
375      lastOp = MatcherOp.FIND;
376      int start;
377      
378      if (includeIndex == -1) {
379        start = 0;
380      } else {
381        start = (delegate.end() == delegate.start()) ? (delegate.end() + 1) : delegate.end();
382      }
383
384      final int regionStart = delegate.regionStart();
385      final int regionEnd = delegate.regionEnd();
386      
387      if (start < regionStart) {
388        start = regionStart;
389      }
390      
391      if (start > regionEnd) {
392        includeIndex = -1;
393        lastStart = -1;
394        return false;
395      }
396
397      lastStart = start;
398      final Set<Interval> rejected = new HashSet<Interval>();
399      int i = -1;
400      
401      for (Pattern pattern : getIncludes()) {
402        i++;
403        
404        if (delegate.usePattern(pattern).find(start)) {
405          final Interval matchedInterval = new Interval(delegate.start(), delegate.end());
406          
407          if (rejected.contains(matchedInterval)) {
408            continue;
409          }
410          
411          if (!matchesExcludes(delegate.start(), delegate.end())) {
412            includeIndex = i;
413            return true;
414          }
415          
416          rejected.add(matchedInterval);
417        }
418      }
419      
420      includeIndex = -1;
421      return false;
422    }
423    
424    private boolean matchesExcludes(final int start, final int end) {
425      if (getExcludes().isEmpty()) {
426        return false;
427      }
428      
429      Matcher matcher = null;
430      
431      for (Pattern pattern : getExcludes()) {
432        @SuppressWarnings("null")
433        @NonNull Pattern nonNullPattern = pattern;
434        
435        if (matcher == null) {
436          matcher = getNewMatcher(nonNullPattern, start, end);
437        } else {
438          matcher.usePattern(nonNullPattern);
439        }
440        
441        if (matcher.matches()) {
442          return true;
443        }
444      }
445      
446      return false;
447    }
448
449    @SuppressWarnings("null")
450    private Matcher getNewMatcher(final Pattern pattern, final int regionStart, final int regionEnd) {
451      return pattern.matcher(input)
452          .region(regionStart, regionEnd)
453          .useTransparentBounds(delegate.hasTransparentBounds())
454          .useAnchoringBounds(delegate.hasAnchoringBounds());
455    }

457    public boolean find(final int start) {
458      lastOp = MatcherOp.FIND_AT;
459      
460      if ((start < 0) || (start > input.length())) {
461        lastStart = -1;
462        includeIndex = -1;
463        throw new IndexOutOfBoundsException("Illegal start index");
464      }
465      
466      lastStart = start;
467      final Set<Interval> rejected = new HashSet<Interval>();
468      int i = -1;
469      
470      for (Pattern pattern : getIncludes()) {
471        i++;
472        
473        if (delegate.usePattern(pattern).find(start)) {
474          final Interval matchedInterval = new Interval(delegate.start(), delegate.end());
475          
476          if (rejected.contains(matchedInterval)) {
477            continue;
478          }
479          
480          if (!matchesExcludes(delegate.start(), delegate.end())) {
481            includeIndex = i;
482            return true;
483          }
484          
485          rejected.add(matchedInterval);
486        }
487      }
488      
489      includeIndex = -1;
490      return false;
491    }
492
493    public boolean lookingAt() {
494      lastOp = MatcherOp.LOOKING_AT;
495      lastStart = -1;
496      final Set<Integer> rejected = new HashSet<Integer>();
497      int i = -1;
498      
499      for (Pattern pattern : getIncludes()) {
500        i++;
501        
502        if (delegate.usePattern(pattern).lookingAt()) {
503          if (rejected.contains(delegate.end())) {
504            continue;
505          }
506          
507          if (!matchesExcludes(delegate.start(), delegate.end())) {
508            includeIndex = i;
509            return true;
510          }
511          
512          rejected.add(delegate.end());
513        }
514      }
515      
516      includeIndex = -1;
517      return false;
518    }

570    public boolean requireEnd() {
571      if (includeIndex == -1) {
572        // requireEnd has no meaning, return false
573        return false; 
574      }

577    }

579    public boolean tryHitEnd() {
580      if (delegate.hitEnd()) {
581        return true;
582      }
583      
584      Matcher matcher = null;
585      int skipIndex = (includeIndex == -1) ? (parentPattern.includeCount() - 1) : includeIndex;
586      int i = -1;
587      
588      for (Pattern pattern : getIncludes()) {
589        i++;
590        
591        if (i == skipIndex) {
592          continue;
593        }
594        
595        @SuppressWarnings("null")
596        @NonNull Pattern nonNullPattern = pattern;
597        
598        if (matcher == null) {
599          matcher = getNewMatcher(nonNullPattern, delegate.regionStart(), delegate.regionEnd());
600        } else {
601          matcher.usePattern(nonNullPattern).reset();
602        }
603        
604        if (lastOp.apply(matcher, lastStart)) {
605          if (matcher.requireEnd()) {
606            return true;
607          }
608        }
609      }
610      
611      return false;
612    }
613
614    public boolean tryRequireEnd() {
615      if (includeIndex == -1) {
616        // requireEnd has no meaning, return false
617        return false; 
618      }
619      
620      if (!delegate.requireEnd()) {
621        return false;
622      }
623      
624      Matcher matcher = null;
625      
626      for (Pattern pattern : getIncludes().subList(includeIndex + 1, getIncludes().size())) {
627        @SuppressWarnings("null")
628        @NonNull Pattern nonNullPattern = pattern;
629        
630        if (matcher == null) {
631          matcher = getNewMatcher(nonNullPattern, delegate.regionStart(), delegate.regionEnd());
632        } else {
633          matcher.usePattern(nonNullPattern).reset();
634        }
635        
636        if (lastOp.apply(matcher, lastStart)) {
637          if (!matchesExcludes(matcher.start(), matcher.end()) && !matcher.requireEnd()) {
638            return false;
639          }
640        }
641      }
642      
643      return true;
644    }

646    private enum MatcherOp {
647      NOOP {
648        @Override
649        public boolean apply(Matcher matcher, int start) {
650          throw new UnsupportedOperationException();
651        }
652      },
653      FIND_AT() {
654        @Override
655        public boolean apply(final Matcher matcher, final int start) {
656          return matcher.find(start);
657        }
658      },
659      LOOKING_AT {
660        @Override
661        public boolean apply(final Matcher matcher, final int start) {
662          return matcher.lookingAt();
663        }
664      },
665      FIND { // internally, same as FIND_AT
666        @Override
667        public boolean apply(final Matcher matcher, final int start) {
668          return matcher.find(start);
669        }
670      },
671      MATCHES {
672        @Override
673        public boolean apply(final Matcher matcher, final int start) {
674          return matcher.matches();
675        }
676      };
677
678      public abstract boolean apply(final Matcher matcher, final int start);
679    }

682  private static final class Interval implements Comparable<Interval>{
683    final int start;
684    final int end;
685    
686    public Interval(int start, int end) {
687      this.start = start;
688      this.end = end;
689    }
690
691    @SuppressWarnings("unused")
692    public int getStart() {
693      return start;
694    }
695
696    @SuppressWarnings("unused")
697    public int getEnd() {
698      return end;
699    }
700    
701    @Override
702    public int hashCode() {
703      final int prime = 31;
704      int result = 1;
705      result = prime * result + end;
706      result = prime * result + start;
707      return result;
708    }
709
710    @Override
711    public boolean equals(@Nullable Object obj) {
712      if (this == obj) {
713        return true;
714      }
715      if (obj == null) {
716        return false;
717      }
718      if (!(obj instanceof Interval)) {
719        return false;
720      }
721      Interval other = (Interval) obj;
722      if (end != other.end) {
723        return false;
724      }
725      if (start != other.start) {
726        return false;
727      }
728      return true;
729    }
730
731    @Override
732    public String toString() {
733      StringBuilder builder = new StringBuilder();
734      builder.append("Interval [start=").append(start).append(", end=").append(end).append("]");
735      @SuppressWarnings("null")
736      @NonNull String str = builder.toString();
737      return str;
738    }
739
740    @Override
741    public int compareTo(final @Nullable Interval other) {
742      if (other == null) {
743        return 1;
744      } else if (start < other.start) {
745        return -1;
746      } else if (start > other.start) {
747        return 1;
748      } else if (end < other.end) {
749        return -1;
750      } else if (end > other.end) {
751        return 1;
752      } else {
753        return 0;
754      }
755    }
