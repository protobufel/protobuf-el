676     * Merges other DynamicMessage into this.
677     *
678     * @see #mergeFrom(Message)
679     */
680    public Builder mergeFrom(final DynamicMessage other) {
681      verifyNotNull(other);
682
683      if (other.getDescriptorForType() != getDescriptorForType()) {
684        throw new IllegalArgumentException(
685            "mergeFrom(Message) can only merge messages of the same type.");
686      }
687
688      if (other.isEmpty()) {
689        return this;
690      }
691
692      for (final Entry<FieldDescriptor, Object> entry : other.getAllFields().entrySet()) {
693        final FieldDescriptor field = entry.getKey();
694
695        if (field.isRepeated()) {
696          final List<?> otherValue = (List<?>) entry.getValue();
697
698          if (!otherValue.isEmpty()) {
699            if (field.getJavaType() == JavaType.MESSAGE) {
700              final RepeatedFieldBuilder<Message, Builder, MessageOrBuilder> fieldBuilder =
701                  getRepeatedFieldBuilder(field, false);
702
703              if (fieldBuilder != null) {
704                @SuppressWarnings("unchecked")
705                final Iterable<? extends Message> otherMessages =
706                (Iterable<? extends Message>) otherValue;
707                fieldBuilder.addAllMessages(otherMessages);
708                continue;
709              }
710            }
711
712            FieldHandlerType.REPEATED_PRIMITIVE.getFieldHandler().addAllRepeatedField(this, field,
713                otherValue);
714          }
715        } else if (field.getJavaType() == JavaType.MESSAGE) {
716          FieldHandlerType.SINGLE_MESSAGE.getFieldHandler().mergeField(this, field,
717              (Message) entry.getValue());
718        } else {
719          builder.setField(field, entry.getValue());
720          onChanged();
721        }
722      }
723
724      mergeUnknownFields(other.getUnknownFields());
725      return this;
726    }

1733    private static void verifyValue(final FieldDescriptor field, final Object value,
1734        final Class<? extends MessageOrBuilder> type) {
1735      if (field.isRepeated()) {
1736        if (value == null) {
1737          throw new NullPointerException();
1738        }
1739
1740        final Iterable<?> values = Iterable.class.cast(value);
1741        final Class<? extends MessageOrBuilder> elementType = type == null ? Message.class : type;
1742
1743        for (final Object element : values) {
1744          verifySingleValue(field, element, elementType);
1745        }
1746      } else {
1747        verifySingleValue(field, value, type);
1748      }

1751    private static void verifySingleValue(final FieldDescriptor field, final Object value,
1752        final Class<? extends MessageOrBuilder> type) {
1753      if (value == null) {
1754        throw new NullPointerException();
1755      }
1756
1757      boolean isValid = false;
1758
1759      switch (field.getJavaType()) {
1760        case INT:
1761          isValid = value instanceof Integer;
1762          break;
1763        case LONG:
1764          isValid = value instanceof Long;
1765          break;
1766        case FLOAT:
1767          isValid = value instanceof Float;
1768          break;
1769        case DOUBLE:
1770          isValid = value instanceof Double;
1771          break;
1772        case BOOLEAN:
1773          isValid = value instanceof Boolean;
1774          break;
1775        case STRING:
1776          isValid = value instanceof String;
1777          break;
1778        case BYTE_STRING:
1779          isValid = value instanceof ByteString;
1780          break;
1781        case ENUM:
1782          if (value instanceof EnumValueDescriptor) {
1783            isValid = ((EnumValueDescriptor) value).getType() == field.getEnumType();
1784          } else if (value instanceof Internal.EnumLite) {
1785            isValid =
1786                field.getEnumType().findValueByNumber(((Internal.EnumLite) value).getNumber()) != null;
1787          } else if (value instanceof Number) {
1788            isValid = field.getEnumType().findValueByNumber(((Number) value).intValue()) != null;
1789          } else if (value instanceof String) {
1790            isValid = field.getEnumType().findValueByName((String) value) != null;
1791          }
1792
1793          break;
1794        case MESSAGE:
1795          final Class<? extends MessageOrBuilder> expectedType =
1796          type == null ? Message.class : (Class<? extends MessageOrBuilder>) type;
1797          final Object resolvedValue =
1798              value instanceof LazyField ? ((LazyField) value).getValue() : value;
1799              isValid =
1800                  expectedType.cast(resolvedValue).getDescriptorForType() == field.getMessageType();
1801              break;
1802      }
1803
1804      if (!isValid) {
1805        // TODO(kenton): When chaining calls to setField(), it can be hard to
1806        // tell from the stack trace which exact call failed, since the whole
1807        // chain is considered one line of code. It would be nice to print
1808        // more information here, e.g. naming the field. We used to do that.
1809        // But we can't now that FieldSet doesn't use descriptors. Maybe this
1810        // isn't a big deal, though, since it would only really apply when using
1811        // reflection and generally people don't chain reflection setters.
1812        throw new IllegalArgumentException(
1813            "Wrong object type used with protocol message reflection.");
1814      }
1815    }

1817    private static Object getVerifiedValue(final FieldDescriptor field, final Object value,
1818        final Class<? extends MessageOrBuilder> type) {
1819      if (field.isRepeated()) {
1820        if (value == null) {
1821          throw new NullPointerException();
1822        }
1823
1824        final Iterable<?> values = Iterable.class.cast(value);
1825        final Class<? extends MessageOrBuilder> elementType = type == null ? Message.class : type;
1826        final List<Object> result = new ArrayList<Object>();
1827
1828        for (final Object element : values) {
1829          result.add(getVerifiedSingleValue(field, element, elementType));
1830        }
1831
1832        return result;
1833      } else {
1834        return getVerifiedSingleValue(field, value, type);
1835      }
1836    }

1838    private static Object getVerifiedSingleValue(final FieldDescriptor field, final Object value,
1839        final Class<? extends MessageOrBuilder> type) {
1840      if (value == null) {
1841        throw new NullPointerException();
1842      }
1843
1844      switch (field.getJavaType()) {
1845        case INT:
1846          return Integer.class.cast(value);
1847        case LONG:
1848          return Long.class.cast(value);
1849        case FLOAT:
1850          return Float.class.cast(value);
1851        case DOUBLE:
1852          return Double.class.cast(value);
1853        case BOOLEAN:
1854          return Boolean.class.cast(value);
1855        case STRING:
1856          return String.class.cast(value);
1857        case BYTE_STRING:
1858          return ByteString.class.cast(value);
1859        case ENUM:
1860          if (value instanceof EnumValueDescriptor) {
1861            if (((EnumValueDescriptor) value).getType() == field.getEnumType()) {
1862              return value;
1863            }
1864          } else {
1865            EnumValueDescriptor enumValue = null;
1866
1867            if (value instanceof Internal.EnumLite) {
1868              enumValue =
1869                  field.getEnumType().findValueByNumber(((Internal.EnumLite) value).getNumber());
1870            } else if (value instanceof Number) {
1871              enumValue = field.getEnumType().findValueByNumber(((Number) value).intValue());
1872            } else if (value instanceof String) {
1873              enumValue = field.getEnumType().findValueByName((String) value);
1874            }
1875
1876            if (enumValue != null) {
1877              return enumValue;
1878            }
1879          }
1880
1881          break;
1882        case MESSAGE:
1883          final Class<? extends MessageOrBuilder> expectedType =
1884          type == null ? Message.class : (Class<? extends MessageOrBuilder>) type;
1885          final MessageOrBuilder resolvedValue =
1886              expectedType
1887              .cast(value instanceof LazyField ? ((LazyField) value).getValue() : value);
1888
1889          if (resolvedValue.getDescriptorForType() == field.getMessageType()) {
1890            return resolvedValue;
1891          }
1892
1893          break;
1894      }
1895
1896      throw new IllegalArgumentException("Wrong object type used with protocol message reflection.");
1897    }
