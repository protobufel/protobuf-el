028package com.github.protobufel.el;
029
030import java.lang.reflect.Constructor;
031import java.lang.reflect.InvocationHandler;
032import java.lang.reflect.Method;
033import java.lang.reflect.Proxy;
034
035/**
036 * EL Reflection Utilities.
037 *
038 * @author protobufel@gmail.com David Tesler
039 */
040public final class ReflectionUtil {
041  private ReflectionUtil() {}
042
043  public interface Invocable {
044    Object invoke(String methodName, Object... args);
045  }
046
047  public interface IDecorator<T, I> {
048    IDecorator<T, I> newInstance();
049
050    void setDelegate(T delegate);
051
052    I get();
053  }
054
055  public static <T, I, E extends IDecorator<T, I>> Object getWrapper(final T original,
056      final Class<I> interfaceClass, final E interfaceImpl, final boolean implementInvokable) {
057    final Class<?>[] interfaces = getAllInterfaces(original, implementInvokable, interfaceClass);
058    final Class<?> proxyClass = Proxy.getProxyClass(interfaceClass.getClassLoader(), interfaces);
059
060    try {
061      final Constructor<?> proxyConstructor =
062          proxyClass.getConstructor(new Class[] {InvocationHandler.class});
063      return proxyConstructor.newInstance(newInvocationHandler(original, interfaceClass,
064          interfaceImpl.newInstance()));
065    } catch (final Throwable e) {
066      throw new RuntimeException(e);
067    }
068  }
069
070  private static <T, I, E extends IDecorator<T, I>> InvocationHandler newInvocationHandler(
071      final T original, final Class<I> interfaceClass, final E interfaceImpl) {
072    interfaceImpl.setDelegate(original);
073    return new InvocationHandler() {
074      @Override
075      public Object invoke(final Object proxy, final Method method, final Object[] args)
076          throws Throwable {
077        if (method.getDeclaringClass() == Invocable.class) {
078          final Object[] invokeArgs = (Object[]) args[1];
079          final Class<?>[] paramTypes = new Class<?>[invokeArgs.length - 1];
080
081          for (int i = 1; i < invokeArgs.length; i++) {
082            paramTypes[i] = invokeArgs[i].getClass();
083          }
084
085          return original.getClass().getMethod((String) args[0], paramTypes)
086              .invoke(original, invokeArgs);
087        } else if (method.getDeclaringClass() == interfaceClass) {
088          return method.invoke(interfaceImpl.get(), args);
089        } else {
090          return method.invoke(proxy, args);
091        }
092      }
093    };
094  }
095
096  private static <T, I, E extends IDecorator<T, I>> InvocationHandler newInvocationHandler2(
097      final T original, final Class<I> interfaceClass, final E interfaceImpl) {
098    interfaceImpl.setDelegate(original);
099    return new DecoratorInvocationHandler<T, I, E>(interfaceClass, interfaceImpl);
100  }
101
102  private static class DecoratorInvocationHandler<T, I, E extends IDecorator<T, I>> implements
103  InvocationHandler {
104    private final Class<I> interfaceClass;
105    private final E interfaceImpl;
106
107    private DecoratorInvocationHandler(final Class<I> interfaceClass, final E interfaceImpl) {
108      this.interfaceClass = interfaceClass;
109      this.interfaceImpl = interfaceImpl;
110    }
111
112    @Override
113    public Object invoke(final Object proxy, final Method method, final Object[] args)
114        throws Throwable {
115      if (method.getDeclaringClass() == Invocable.class) {
116        final Object[] invokeArgs = (Object[]) args[1];
117        final Class<?>[] paramTypes = new Class<?>[invokeArgs.length - 1];
118
119        for (int i = 1; i < invokeArgs.length; i++) {
120          paramTypes[i] = invokeArgs[i].getClass();
121        }
122
123        return proxy.getClass().getMethod((String) args[0], paramTypes).invoke(proxy, invokeArgs);
124      } else if (method.getDeclaringClass() == interfaceClass) {
125        return method.invoke(interfaceImpl.get(), args);
126      } else {
127        return method.invoke(proxy, args);
128      }
129    }
130  }
131
132  public interface ConditionalInvocationHandler extends InvocationHandler {
133    boolean isApplicable(Object proxy, Method method, Object[] args);
134  }
135
136  private static class InvocationHandlerDecorator<T> implements InvocationHandler {
137    private final T original;
138    private final InvocationHandler handler;
139
140    public InvocationHandlerDecorator(final T original, final InvocationHandler handler) {
141      this.original = original;
142      this.handler = handler;
143    }
144
145    @Override
146    public Object invoke(final Object proxy, final Method method, final Object[] args)
147        throws Throwable {
148      if (!method.getDeclaringClass().isAssignableFrom(original.getClass())
149          || handler instanceof ConditionalInvocationHandler
150          && ((ConditionalInvocationHandler) handler).isApplicable(original, method, args)) {
151        return handler.invoke(original, method, args);
152      } else {
153        return method.invoke(original, args);
154      }
155    }
156  }
157
158  private static Class<?>[] getAllInterfaces(final Object object, final boolean implementInvokable,
159      final Class<?>... extraInterfaces) {
160    final Class<?> clazz = object == null ? Object.class : object.getClass();
161    Class<?>[] result;
162    final int addInvokable = implementInvokable ? 1 : 0;
163
164    if (clazz.isInterface()) {
165      result = new Class<?>[extraInterfaces.length + 1 + addInvokable];
166      System.arraycopy(extraInterfaces, 0, result, 1, extraInterfaces.length);
167      result[0] = clazz;
168    } else {
169      final Class<?>[] interfaces = clazz.getInterfaces();
170      result = new Class<?>[extraInterfaces.length + interfaces.length + addInvokable];
171      System.arraycopy(interfaces, 0, result, 0, interfaces.length);
172      System.arraycopy(extraInterfaces, 0, result, interfaces.length, extraInterfaces.length);
173    }
174
175    if (implementInvokable) {
176      result[result.length - 1] = Invocable.class;
177    }
178
179    return result;
180  }
