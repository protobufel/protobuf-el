045  public static String[] split(IPatternSupport pattern, CharSequence input, int limit) {
046    int index = 0;
047    boolean matchLimited = limit > 0;
048    final ArrayList<String> matchList = new ArrayList<String>();
049    final IMatcherSupport m = pattern.matcher(input);
050
051    // Add segments before each match found
052    while (m.find()) {
053      if (!matchLimited || matchList.size() < limit - 1) {
054        String match = input.subSequence(index, m.start()).toString();
055        matchList.add(match);
056        index = m.end();
057      } else if (matchList.size() == limit - 1) { // last one
058        String match = input.subSequence(index, input.length()).toString();
059        matchList.add(match);
060        index = m.end();
061      }
062    }
063
064    // If no match was found, return this
065    if (index == 0) {
066      return new String[] { input.toString() };
067    }
068
069    // Add remaining segment
070    if (!matchLimited || matchList.size() < limit) {
071      matchList.add(input.subSequence(index, input.length()).toString());
072    }
073
074    // Construct result
075    int resultSize = matchList.size();
076
077    if (limit == 0) {
078      while (resultSize > 0 && matchList.get(resultSize - 1).equals("")) {
079        resultSize--;
080      }
081    }
082
083    String[] result = new String[resultSize];
084    @SuppressWarnings("null")
085    @NonNull String[] array = matchList.subList(0, resultSize).toArray(result);
086    return array;
087  }

089  public static <T extends IMatcherSupport> T appendReplacement(T matcher, StringBuffer sb,
090      String replacement) {
091    final int lastAppendPosition = appendReplacement(matcher.getDelegate(), matcher.getInput(), sb,
092        replacement, matcher.getLastAppendPosition());
093    matcher.setLastAppendPosition(lastAppendPosition);
094    return matcher;
095  }

097  public static int appendReplacement(Matcher matcher, CharSequence text, StringBuffer sb,
098      String replacement, int lastAppendPosition) {
099    matcher.start(); // will throw an exception if no match
100    // Process substitution string to replace group references with groups
101    int cursor = 0;
102    StringBuilder result = new StringBuilder();
103
104    while (cursor < replacement.length()) {
105      char nextChar = replacement.charAt(cursor);
106
107      if (nextChar == '\\') {
108        cursor++;
109        nextChar = replacement.charAt(cursor);
110        result.append(nextChar);
111        cursor++;
112      } else if (nextChar == '$') {
113        // Skip past $
114        cursor++;
115        // A StringIndexOutOfBoundsException is thrown if
116        // this "$" is the last character in replacement
117        // string in current implementation, a IAE might be
118        // more appropriate.
119        nextChar = replacement.charAt(cursor);
120        int refNum = -1;
121
122        if (nextChar == '{') {
123          cursor++;
124          StringBuilder gsb = new StringBuilder();
125
126          while (cursor < replacement.length()) {
127            nextChar = replacement.charAt(cursor);
128
129            if (ASCII.isLower(nextChar) || ASCII.isUpper(nextChar) || ASCII.isDigit(nextChar)) {
130              gsb.append(nextChar);
131              cursor++;
132            } else {
133              break;
134            }
135          }
136
137          if (gsb.length() == 0) {
138            throw new IllegalArgumentException("named capturing group has 0 length name");
139          }
140
141          if (nextChar != '}') {
142            throw new IllegalArgumentException("named capturing group is missing trailing '}'");
143          }
144
145          String gname = gsb.toString();
146
147          if (ASCII.isDigit(gname.charAt(0))) {
148            throw new IllegalArgumentException("capturing group name {" + gname
149                + "} starts with digit character");
150          }
151
152          final String group = matcher.group(gname);
153
154          // Append group
155          if (group != null) {
156            result.append(group);
157          }
158
159          cursor++;
160        } else {
161          // The first number is always a group
162          refNum = (int) nextChar - '0';
163          if ((refNum < 0) || (refNum > 9))
164            throw new IllegalArgumentException("Illegal group reference");
165          cursor++;
166          // Capture the largest legal group string
167          boolean done = false;
168          while (!done) {
169            if (cursor >= replacement.length()) {
170              break;
171            }
172            int nextDigit = replacement.charAt(cursor) - '0';
173            if ((nextDigit < 0) || (nextDigit > 9)) { // not a number
174              break;
175            }
176            int newRefNum = (refNum * 10) + nextDigit;
177            if (matcher.groupCount() < newRefNum) {
178              done = true;
179            } else {
180              refNum = newRefNum;
181              cursor++;
182            }
183          }
184
185          // Append group
186          if (matcher.start(refNum) != -1 && matcher.end(refNum) != -1) {
187            result.append(text, matcher.start(refNum), matcher.end(refNum));
188          }
189        }
190      } else {
191        result.append(nextChar);
192        cursor++;
193      }
194    }
195
196    // Append the intervening text
197    sb.append(text, lastAppendPosition, matcher.start());
198    // Append the match substitution
199    sb.append(result);
200
201    return matcher.end();
202  }

204  public static StringBuffer appendTail(CharSequence text, int lastAppendPosition, 
205      StringBuffer sb) {
206    sb.append(text, lastAppendPosition, text.length());
207    return sb;
208  }

210  public static String replaceAll(IMatcherSupport matcher, String replacement) {
211    matcher.reset();
212    boolean result = matcher.find();
213    final CharSequence text = matcher.getInput();
214
215    if (result) {
216      StringBuffer sb = new StringBuffer();
217
218      do {
219        final int lastAppendPosition = appendReplacement(matcher.getDelegate(), text, sb,
220            replacement, matcher.getLastAppendPosition());
221        matcher.setLastAppendPosition(lastAppendPosition);
222        result = matcher.find();
223      } while (result);
224
225      appendTail(text, matcher.getLastAppendPosition(), sb);
226      @SuppressWarnings("null")
227      @NonNull String str = sb.toString();
228      return str;
229    }
230    
231    @SuppressWarnings("null")
232    @NonNull String str = text.toString();
233    return str;
234  }

236  public static String replaceFirst(IMatcherSupport matcher, String replacement) {
237//    if (replacement == null) {
238//      throw new NullPointerException("replacement");
239//    }
240
241    final CharSequence text = matcher.getInput();
242    matcher.reset();
243
244    if (!matcher.find()) {
245      @SuppressWarnings("null")
246      @NonNull String str = text.toString();
247      return str;
248    }
249
250    StringBuffer sb = new StringBuffer();
251    final int lastAppendPosition = appendReplacement(matcher.getDelegate(), text, sb, replacement,
252        matcher.getLastAppendPosition());
253    matcher.setLastAppendPosition(lastAppendPosition);
254
255    appendTail(text, matcher.getLastAppendPosition(), sb);
256    @SuppressWarnings("null")
257    @NonNull String str = sb.toString();
258    return str;
259  }

430    static int getType(int ch) {
431      return ((ch & 0xFFFFFF80) == 0 ? ctype[ch] : 0);
432    }
433
434    static boolean isType(int ch, int type) {
435      return (getType(ch) & type) != 0;
436    }
437
438    static boolean isAscii(int ch) {
439      return ((ch & 0xFFFFFF80) == 0);
440    }
441
442    static boolean isAlpha(int ch) {
443      return isType(ch, ALPHA);
444    }
445
446    static boolean isDigit(int ch) {
447      return ((ch - '0') | ('9' - ch)) >= 0;
448    }
449
450    static boolean isAlnum(int ch) {
451      return isType(ch, ALNUM);
452    }
453
454    static boolean isGraph(int ch) {
455      return isType(ch, GRAPH);
456    }

458    static boolean isPrint(int ch) {
