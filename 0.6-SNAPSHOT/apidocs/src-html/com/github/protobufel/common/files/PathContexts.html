028package com.github.protobufel.common.files;
029
030import static com.github.protobufel.common.verifications.Verifications.verifyNonNull;
031
032import java.io.File;
033import java.nio.file.Path;
034
035import org.eclipse.jdt.annotation.NonNull;
036import org.eclipse.jdt.annotation.Nullable;
037
038public class PathContexts {
039  private static final PathContext<Object> EMPTY_PATH_CONTEXT = new PathContext<Object>() {
040    @Override
041    public int currentDepth() {
042      return -1;
043    }
044
045    @Override
046    public boolean setCachedValue(Object key, Object value) {
047      return false;
048    }
049
050    @Override
051    public @Nullable Object getCachedValue(Object key) {
052      return null;
053    }
054
055    @Override
056    public @Nullable String resolvePath(Object basePath, Object path) {
057      return null;
058    }
059
060    @Override
061    public @Nullable String resolvePath(Object path) {
062      return null;
063    }
064
065    @Override
066    public @Nullable String getSeparator(Object path) {
067      return null;
068    }
069  };
070  
071  private PathContexts() {}
072  
073  public interface PathContext<T> extends IHistoryCacheView<Object, Object> {
074    @Nullable String resolvePath(T basePath, T path);
075    @Nullable String resolvePath(T path);
076    @Nullable String getSeparator(T path);
077    //void reportError(String message, String path);
078  }
079
080  @SuppressWarnings("unchecked")
081  public static <T> PathContext<T> emptyPathContext() {
082    return (PathContext<T>) EMPTY_PATH_CONTEXT;
083  }
084  
085  public static class BasePathContext<T> implements PathContext<T> {
086    private final IHistoryCacheView<Object, Object> cache;
087    private final @Nullable T root;
088    
089    public BasePathContext(IHistoryCacheView<Object, Object> cache, @Nullable T root) {
090      this.cache = verifyNonNull(cache);
091      this.root = root;
092    }
093
094    protected @Nullable T getRoot() {
095      return root;
096    }
097
098    @Override
099    public @Nullable String resolvePath(T path) {
100      final T root = this.root;
101      
102      if (root != null) {
103        return resolvePath(root, path);
104      } else {
105        return path.toString();
106      }
107    }
108
109    @Override
110    public @Nullable String resolvePath(T basePath, T path) {
111      final @NonNull String strPath = verifyNonNull(path.toString());
112      final @NonNull String strBasePath = verifyNonNull(basePath.toString()) + File.separator;
113      return strPath.startsWith(strBasePath) ? strPath.substring(strBasePath.length()) : null;
114    }
115
116    @Override
117    public @Nullable String getSeparator(T path) {
118      return File.separator;
119    }
120
121    @Override
122    public @Nullable Object getCachedValue(Object key) {
123      return cache.getCachedValue(key);
124    }
125
126    @Override
127    public boolean setCachedValue(Object key, Object value) {
128      return cache.setCachedValue(key, value);
129    }
130
131    @Override
132    public int currentDepth() {
133      return cache.currentDepth();
134    }
135  }
136
137  public static final class SimplePathContext extends BasePathContext<Path> {
138
139    public SimplePathContext(IHistoryCacheView<Object, Object> cache, @Nullable Path root) {
140      super(cache, root);
141    }
142
143    @Override
144    public @Nullable String resolvePath(Path basePath, Path path) {
145      return path.startsWith(basePath) ? basePath.relativize(path).toString() : null;
146    }
147
148    @Override
149    public @Nullable String resolvePath(Path path) {
150      final Path root = getRoot();
151      
152      if (root != null) {
153        return resolvePath(root, path);
154      } else {
155        return path.toString();
156      }
157    }
158
159    @Override
160    public @Nullable String getSeparator(Path path) {
161      return path.getFileSystem().getSeparator();
162    }
163  }
164}
