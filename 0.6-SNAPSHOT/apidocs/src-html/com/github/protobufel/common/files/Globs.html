026package com.github.protobufel.common.files;
027
028import java.util.regex.PatternSyntaxException;
029
030/**
031 * 
032 * An utility for a glob to regex conversion. 
033 * This is a near mirror of the OpenJDK 7 {@code sun.nio.fs.Globs} with some 
034 * methods made public, and the returned Pattern without first ^ and last $.
035 * 
036 * @author protobufel@gmail.com David Tesler
037 */
038public class Globs {
039    private Globs() { }
040
041    private static final String regexMetaChars = ".^$+{[]|()";
042    private static final String globMetaChars = "\\*?[{";
043
044    private static boolean isRegexMeta(char c) {
045        return regexMetaChars.indexOf(c) != -1;
046    }
047
048    private static boolean isGlobMeta(char c) {
049        return globMetaChars.indexOf(c) != -1;
050    }
051    private static char EOL = 0;  //TBD
052
053    private static char next(String glob, int i) {
054        if (i < glob.length()) {
055            return glob.charAt(i);
056        }
057        return EOL;
058    }
059
060    /**
061     * Creates a regex pattern from the given glob expression.
062     *
063     * @throws  PatternSyntaxException
064     */
065    public static String toRegexPattern(String globPattern, boolean isDos) {
066        boolean inGroup = false;
067        // StringBuilder regex = new StringBuilder("^"); - original
068        StringBuilder regex = new StringBuilder();
069
070        int i = 0;
071        while (i < globPattern.length()) {
072            char c = globPattern.charAt(i++);
073            switch (c) {
074                case '\\':
075                    // escape special characters
076                    if (i == globPattern.length()) {
077                        throw new PatternSyntaxException("No character to escape",
078                                globPattern, i - 1);
079                    }
080                    char next = globPattern.charAt(i++);
081                    if (isGlobMeta(next) || isRegexMeta(next)) {
082                        regex.append('\\');
083                    }
084                    regex.append(next);
085                    break;
086                case '/':
087                    if (isDos) {
088                        regex.append("\\\\");
089                    } else {
090                        regex.append(c);
091                    }
092                    break;
093                case '[':
094                    // don't match name separator in class
095                    if (isDos) {
096                        regex.append("[[^\\\\]&&[");
097                    } else {
098                        regex.append("[[^/]&&[");
099                    }
100                    if (next(globPattern, i) == '^') {
101                        // escape the regex negation char if it appears
102                        regex.append("\\^");
103                        i++;
104                    } else {
105                        // negation
106                        if (next(globPattern, i) == '!') {
107                            regex.append('^');
108                            i++;
109                        }
110                        // hyphen allowed at start
111                        if (next(globPattern, i) == '-') {
112                            regex.append('-');
113                            i++;
114                        }
115                    }
116                    boolean hasRangeStart = false;
117                    char last = 0;
118                    while (i < globPattern.length()) {
119                        c = globPattern.charAt(i++);
120                        if (c == ']') {
121                            break;
122                        }
123                        if (c == '/' || (isDos && c == '\\')) {
124                            throw new PatternSyntaxException("Explicit 'name separator' in class",
125                                    globPattern, i - 1);
126                        }
127                        // TBD: how to specify ']' in a class?
128                        if (c == '\\' || c == '[' ||
129                                c == '&' && next(globPattern, i) == '&') {
130                            // escape '\', '[' or "&&" for regex class
131                            regex.append('\\');
132                        }
133                        regex.append(c);
134
135                        if (c == '-') {
136                            if (!hasRangeStart) {
137                                throw new PatternSyntaxException("Invalid range",
138                                        globPattern, i - 1);
139                            }
140                            if ((c = next(globPattern, i++)) == EOL || c == ']') {
141                                break;
142                            }
143                            if (c < last) {
144                                throw new PatternSyntaxException("Invalid range",
145                                        globPattern, i - 3);
146                            }
147                            regex.append(c);
148                            hasRangeStart = false;
149                        } else {
150                            hasRangeStart = true;
151                            last = c;
152                        }
153                    }
154                    if (c != ']') {
155                        throw new PatternSyntaxException("Missing ']", globPattern, i - 1);
156                    }
157                    regex.append("]]");
158                    break;
159                case '{':
160                    if (inGroup) {
161                        throw new PatternSyntaxException("Cannot nest groups",
162                                globPattern, i - 1);
163                    }
164                    regex.append("(?:(?:");
165                    inGroup = true;
166                    break;
167                case '}':
168                    if (inGroup) {
169                        regex.append("))");
170                        inGroup = false;
171                    } else {
172                        regex.append('}');
173                    }
174                    break;
175                case ',':
176                    if (inGroup) {
177                        regex.append(")|(?:");
178                    } else {
179                        regex.append(',');
180                    }
181                    break;
182                case '*':
183                    if (next(globPattern, i) == '*') {
184                        // crosses directory boundaries
185                        regex.append(".*");
186                        i++;
187                    } else {
188                        // within directory boundary
189                        if (isDos) {
190                            regex.append("[^\\\\]*");
191                        } else {
192                            regex.append("[^/]*");
193                        }
194                    }
195                    break;
196                case '?':
197                   if (isDos) {
198                       regex.append("[^\\\\]");
199                   } else {
200                       regex.append("[^/]");
201                   }
202                   break;
203
204                default:
205                    if (isRegexMeta(c)) {
206                        regex.append('\\');
207                    }
208                    regex.append(c);
209            }
210        }
211
212        if (inGroup) {
213            throw new PatternSyntaxException("Missing '}", globPattern, i - 1);
214        }
215
216        // return regex.append('$').toString(); - original
217        return regex.toString();
218    }
219
220    public static String toUnixRegexPattern(String globPattern) {
221        return toRegexPattern(globPattern, false);
222    }
223
224    public static String toWindowsRegexPattern(String globPattern) {
225        return toRegexPattern(globPattern, true);
226    }
227}
