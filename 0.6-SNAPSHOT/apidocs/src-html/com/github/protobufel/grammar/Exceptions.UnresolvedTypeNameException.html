028package com.github.protobufel.grammar;
029
030import java.util.List;
031
032import com.google.protobuf.Descriptors.DescriptorValidationException;
033import com.google.protobuf.Message;
034
035/**
036 * Various validation exceptions used by {@link ProtoFileParser}.
037 *
038 * @author protobufel@gmail.com David Tesler
039 */
040public final class Exceptions {
041  private Exceptions() {}
042
043  public static interface IProtoParserException {
044    public String getMessageFormat();
045  }
046
047  public static final class NonUniqueException extends RuntimeException implements
048      IProtoParserException {
049    private static final long serialVersionUID = 1L;
050    private static final String MESSAGE_FORMAT = "the %s:%s already set";
051    private final String propertyName;
052
053    public NonUniqueException(final Object value, final String propertyName) {
054      super(String.format(MESSAGE_FORMAT, propertyName, value));
055      this.propertyName = propertyName + ":" + value;
056    }
057
058    public String getPropertyName() {
059      return propertyName;
060    }
061
062    @Override
063    public String getMessageFormat() {
064      return MESSAGE_FORMAT;
065    }
066  }
067
068  public static final class NonUniqueExtensionNumber extends RuntimeException implements
069      IProtoParserException {
070    private static final long serialVersionUID = 1L;
071    private static final String MESSAGE_FORMAT = "extension '%s.%s' number %d is already taken";
072
073    public NonUniqueExtensionNumber(final String message) {
074      super(message);
075    }
076
077    public NonUniqueExtensionNumber(final String extendee, final String name, final int number) {
078      super(String.format(MESSAGE_FORMAT, extendee, name, number));
079    }
080
081    @Override
082    public String getMessageFormat() {
083      return MESSAGE_FORMAT;
084    }
085  }
086
087  public static final class FieldInExtensionRangeException extends RuntimeException implements
088      IProtoParserException {
089    private static final long serialVersionUID = 1L;
090    private static final String MESSAGE_FORMAT =
091        "the %s field number should not be in the message/group extension ranges";
092
093    public FieldInExtensionRangeException(final String fieldName) {
094      super(String.format(MESSAGE_FORMAT, fieldName));
095    }
096
097    @Override
098    public String getMessageFormat() {
099      return MESSAGE_FORMAT;
100    }
101  }
102
103  public static final class InvalidExtensionRange extends RuntimeException implements
104      IProtoParserException {
105    private static final long serialVersionUID = 1L;
106    private static final String MESSAGE_FORMAT = "extension range[%s, %s] overlaps with others";
107
108    public InvalidExtensionRange(final int start, final int end) {
109      super(String.format(MESSAGE_FORMAT, start, end));
110    }
111
112    public InvalidExtensionRange(final String message) {
113      super(message);
114    }
115
116    @Override
117    public String getMessageFormat() {
118      return MESSAGE_FORMAT;
119    }
120  }
121
122  public static final class UnresolvedTypeNameException extends RuntimeException implements
123      IProtoParserException {
124    private static final long serialVersionUID = 1L;
125    private static final String MESSAGE_FORMAT = "The field %s has unresolvable type name: %s";
126
127    public UnresolvedTypeNameException(final String fieldName, final List<?> unresolvedInfo) {
128      super(String.format(MESSAGE_FORMAT, fieldName, unresolvedInfo));
129    }
130
131    @Override
132    public String getMessageFormat() {
133      return MESSAGE_FORMAT;
134    }
135  }
136
137  public static final class CircularProtoDependenciesException extends RuntimeException implements
138      IProtoParserException {
139    private static final long serialVersionUID = 1L;
140    private static final String MESSAGE_FORMAT =
141        "these FileDescriprorProtos are circular or refer to nonexisting files: %s";
142    private final List<String> protoNames;
143
144    public CircularProtoDependenciesException(final List<String> protoNames) {
145      super(String.format(MESSAGE_FORMAT, protoNames));
146      this.protoNames = protoNames;
147    }
148
149    public List<String> getProtoNames() {
150      return protoNames;
151    }
152
153    @Override
154    public String getMessageFormat() {
155      return MESSAGE_FORMAT;
156    }
157  }
158
159  public static final class DescriptorValidationRuntimeException extends RuntimeException {
160    public static final String REPEATING_CUSTOM_OPTION_SINGULAR_FIELD_MSG =
161        "repeating custom option's singular field";
162    private static final long serialVersionUID = 1L;
163    private final String problemSymbolName;
164    private final Message proto;
165    private final String description;
166
167    public DescriptorValidationRuntimeException(final String problemSymbolName,
168        final Message proto, final String description) {
169      super(problemSymbolName + ": " + description);
170      this.problemSymbolName = problemSymbolName;
171      this.proto = proto;
172      this.description = description;
173    }
174
175    public DescriptorValidationRuntimeException(final DescriptorValidationException delegate) {
176      super(delegate.getMessage(), delegate);
177      problemSymbolName = delegate.getProblemSymbolName();
178      proto = delegate.getProblemProto();
179      description = delegate.getDescription();
180    }
181
182    public String getProblemSymbolName() {
183      return problemSymbolName;
184    }
185
186    public Message getProblemProto() {
187      return proto;
188    }
189
190    public String getDescription() {
191      return description;
192    }
193  }
194}
