028package com.github.protobufel.common.files;
029
030import static com.github.protobufel.common.verifications.Verifications.*;
031
032import java.util.regex.Matcher;
033import java.util.regex.Pattern;
034
035import org.eclipse.jdt.annotation.NonNull;
036import org.eclipse.jdt.annotation.NonNullByDefault;
037import org.eclipse.jdt.annotation.Nullable;
038
039import com.github.protobufel.common.files.PathContexts.PathContext;
040
041public final class ContextPathMatchers {
042  private ContextPathMatchers() {}
043
044  interface HierarchicalMatcherCommon {
045    boolean isEmpty();
046    boolean isAllowDirs();
047    boolean isAllowFiles();
048    String getPattern();
049  }
050  
051  public interface ContextHierarchicalMatcher<T> extends HierarchicalMatcherCommon {
052    boolean matches(T path, PathContext<T> pathContext);
053    DirectoryMatchResult matchesDirectory(T path, PathContext<T> pathContext);
054  }
055
056  public interface BasicHierarchicalMatcher<T> extends HierarchicalMatcherCommon {
057    boolean matchesResolved(@Nullable String path, PathContext<T> pathContext);
058    DirectoryMatchResult matchesResolvedDirectory(@Nullable String path, @Nullable String separator, 
059        PathContext<T> pathContext);
060  }
061
062  public interface HierarchicalMatcher<T> extends ContextHierarchicalMatcher<T>, BasicHierarchicalMatcher<T> {
063  }
064  
065  public static <T> HierarchicalMatcher<T> getHierarchicalMatcher(final String syntaxAndPattern, 
066      final boolean isUnix, final boolean allowDirs, final boolean allowFiles, 
067      final Class<? extends T> pathType) {
068    final String[] parts = verifyNonNull(syntaxAndPattern).split(":", 2);
069    final @NonNull String regex = verifyArgument(parts.length == 2, parts[1]);
070    
071    if (parts[0].equals("regex")) {
072      return new SimpleHierarchicalMatcher<T>(isUnix, regex, allowDirs, allowFiles);
073    } else if (parts[0].equals("glob")) {
074      return new SimpleHierarchicalMatcher<T>(regex , isUnix, allowDirs, allowFiles);
075    } else {
076      throw new UnsupportedOperationException(String.format("%s syntax", parts[0]));
077    }
078  }
079  
080  public static final class SimpleHierarchicalMatcher<T> implements HierarchicalMatcher<T> {
081    private final Pattern pattern;
082    private final boolean allowDirs;
083    private final boolean allowFiles;
084    private final boolean isUnix;
085
086    public SimpleHierarchicalMatcher(final SimpleHierarchicalMatcher<T> other) {
087      this.pattern = other.pattern;
088      this.allowDirs = other.allowDirs;
089      this.allowFiles = other.allowFiles;
090      this.isUnix = other.isUnix;
091    }
092
093    public SimpleHierarchicalMatcher(final String glob, final boolean isUnix, final boolean allowDirs, 
094        final boolean allowFiles) {
095      this(isUnix, convertGlobToRegex(glob, isUnix), true, allowDirs, allowFiles);
096    }
097
098    public SimpleHierarchicalMatcher(final boolean isUnix, final String regex, 
099        final boolean allowDirs, final boolean allowFiles) {
100      this(isUnix, regex, false, allowDirs, allowFiles);
101    }
102
103    public SimpleHierarchicalMatcher(final boolean isUnix, final String regex, 
104        final boolean isRegexSystemSpecific, final boolean allowDirs, final boolean allowFiles) {
105      verifyCondition((allowDirs || allowFiles), "allowDirs and allowFiles cannot be both false");
106      this.allowDirs = allowDirs;
107      this.allowFiles = allowFiles;
108      final String regexFlags = isUnix ? "" : "(?iu)";
109      final String fsRegex = isRegexSystemSpecific ? regex 
110          : convertRegexToSystemSpecific(regex, isUnix);
111      @SuppressWarnings("null")
112      @NonNull Pattern compiled = Pattern.compile(regexFlags + verifyNonNull(fsRegex));
113      this.pattern = compiled;
114      this.isUnix = isUnix;
115    }
116
117    @SuppressWarnings("null")
118    @Override
119    public String getPattern() {
120      return pattern.pattern();
121    }
122    
123    @Override
124    public String toString() {
125      return "SimpleHierarchicalMatcher [pattern=" + pattern + ", flags=" + pattern.flags() 
126          + ", allowDirs=" + allowDirs + ", allowFiles=" + allowFiles + ", isUnix=" + isUnix + "]";
127    }
128    
129    @Override
130    public int hashCode() {
131      final int prime = 31;
132      int result = 1;
133      result = prime * result + (allowDirs ? 1231 : 1237);
134      result = prime * result + (allowFiles ? 1231 : 1237);
135      result = prime * result + (isUnix ? 1231 : 1237);
136      result = prime * result + pattern.pattern().hashCode();
137      return result;
138    }
139
140    @Override
141    @NonNullByDefault(false)
142    public boolean equals(Object obj) {
143      if (this == obj) {
144        return true;
145      }
146      if (obj == null) {
147        return false;
148      }
149      if (!(obj instanceof SimpleHierarchicalMatcher)) {
150        return false;
151      }
152      final SimpleHierarchicalMatcher<?> other = (SimpleHierarchicalMatcher<?>) obj;
153      if (allowDirs != other.allowDirs) {
154        return false;
155      }
156      if (allowFiles != other.allowFiles) {
157        return false;
158      }
159      if (isUnix != other.isUnix) {
160        return false;
161      }
162      if (!pattern.pattern().equals(other.pattern.pattern())) {
163        return false;
164      }
165      return true;
166    }
167
168    @Override
169    public boolean isEmpty() {
170      return "".equals(pattern.pattern());
171    }
172
173    public boolean isAllowDirs() {
174      return allowDirs;
175    }
176
177    public boolean isAllowFiles() {
178      return allowFiles;
179    }
180
181    @Override
182    public boolean matches(final T path, final PathContext<T> pathContext) {
183      return matchesResolved(pathContext.resolvePath(path), pathContext);
184    }
185
186    @Override
187    public DirectoryMatchResult matchesDirectory(final T path, final PathContext<T> pathContext) {
188      return matchesResolvedDirectory(pathContext.resolvePath(path), 
189          pathContext.getSeparator(path), pathContext);
190    }
191
192    @Override
193    public boolean matchesResolved(final @Nullable String path, final PathContext<T> pathContext) {
194      return (allowFiles && (path != null)) ? pattern.matcher(path).matches() : false;
195    }
196
197    @Override
198    public DirectoryMatchResult matchesResolvedDirectory(final @Nullable String path, 
199        final @Nullable String separator, final PathContext<T> pathContext) {
200      if ((path == null) || (separator == null)) {
201        return DirectoryMatchResult.NO_MATCH;
202      }
203      
204      final String pathWithSlash = path + separator;
205      final Matcher matcher;
206      final boolean isMatched;
207      
208      if (allowDirs) {
209        matcher = pattern.matcher(path);
210        isMatched = matcher.matches();
211        matcher.reset(pathWithSlash);
212      } else {
213        matcher = pattern.matcher(pathWithSlash);
214        isMatched = false;
215      }
216      
217      final boolean skip = matcher.matches() ? matcher.requireEnd() : (!matcher.hitEnd());
218      return DirectoryMatchResult.valueOf(isMatched, skip);
219    }
220    
221    protected final static String convertGlobToRegex(final String glob, final boolean isUnix) {
222      if (isUnix) {
223        return Globs.toUnixRegexPattern(glob);
224      } else {
225        return Globs.toWindowsRegexPattern(glob);
226      }
227    }
228    
229    protected final String convertRegexToSystemSpecific(final String normalizedRelativeRegex, 
230        final boolean isUnix) {
231      if (isUnix) {
232        return normalizedRelativeRegex;
233      } else {
234        return convertNormalizedRelativeRegexToWindows(normalizedRelativeRegex);
235      }
236    }
237    
238    @SuppressWarnings("null")
239    private static final Pattern UNIX_TO_WINDOWS_SLASH_FIND_REGEX = Pattern
240        .compile("/|(?:(\\\\*+)(?:(\\[)|(\\])|(Q)|(E)))");
241    
242    protected final String convertNormalizedRelativeRegexToWindows(
243        final String normalizedRelativeRegex) {
244      @SuppressWarnings("null")
245      final @NonNull Matcher matcher = UNIX_TO_WINDOWS_SLASH_FIND_REGEX.matcher(
246          normalizedRelativeRegex);
247      final StringBuffer sb = new StringBuffer();
248      boolean foundCharGroup = false;
249      boolean foundQuote = false;
250
251      while (matcher.find()) {
252        if (foundQuote) {
253          if (matcher.group(5) != null) {
254            if ((matcher.group(1).length() % 2) == 1) {
255              foundQuote = false;
256            }
257          }
258        } else if (matcher.group(4) != null) {
259          if ((matcher.group(1).length() % 2) == 1) {
260            foundQuote = true;
261          }
262        } else if (matcher.group(2) != null) {
263          if ((matcher.group(1).length() % 2) == 0) {
264            foundCharGroup = true;
265          }
266        } else if (matcher.group(3) != null) {
267          if ((matcher.group(1).length() % 2) == 0) {
268            foundCharGroup = false;
269          }
270        } else if (!foundCharGroup) {
271          matcher.appendReplacement(sb, "\\\\\\\\");
272        }
273      }
274
275      @SuppressWarnings("null")
276      final @NonNull String result = matcher.appendTail(sb).toString();
277      return result;
278    }
279  }
